97 Display_Handle DisplayUart_open(Display_Handle hDisplay,
98  Display_Params *params)
99 {
100  DisplayUart_HWAttrs *hwAttrs = (DisplayUart_HWAttrs *)hDisplay->hwAttrs;
101  DisplayUart_Object *object = (DisplayUart_Object *)hDisplay->object;
102 
103  UART_Params uartParams;
104  UART_Params_init(&uartParams);
105  uartParams.baudRate = hwAttrs->baudRate;
106  uartParams.writeMode = UART_MODE_BLOCKING;
107 
108  Semaphore_Params semParams;
109  Semaphore_Params_init(&semParams);
110  semParams.mode = Semaphore_Mode_BINARY;
111  Semaphore_construct(&object->mutex, 1, &semParams);
112 
113  object->hUart = UART_open(hwAttrs->uartIdx, &uartParams);
114  if (NULL == object->hUart)
115  {
116  Log_print0(Diags_USER1, "DisplayUart.c: Couldn't open UART");
117  return NULL;
118  }
119 
120  return hDisplay;
121 }

168 void DisplayUart_put5(Display_Handle hDisplay, uint8_t line,
169  uint8_t column, uintptr_t fmt, uintptr_t a0,
170  uintptr_t a1, uintptr_t a2, uintptr_t a3, uintptr_t a4)
171 {
172  DisplayUart_Object *object = (DisplayUart_Object *)hDisplay->object;
173  DisplayUart_HWAttrs *hwAttrs = (DisplayUart_HWAttrs *)hDisplay->hwAttrs;
174 
175  uint32_t strSize = 0;
176 
177  if (Semaphore_pend((Semaphore_Handle) & object->mutex, hwAttrs->mutexTimeout))
178  {
179  System_snprintf(hwAttrs->strBuf, hwAttrs->strBufLen - 2, (xdc_CString)fmt, a0, a1, a2, a3, a4);
180 
181  strSize = strlen(hwAttrs->strBuf);
182  hwAttrs->strBuf[strSize++] = '\r';
183  hwAttrs->strBuf[strSize++] = '\n';
184 
185  UART_write(object->hUart, hwAttrs->strBuf, strSize);
186  Semaphore_post((Semaphore_Handle) & object->mutex);
187  }
188 }
