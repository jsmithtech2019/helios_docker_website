212 * @param none

217 {

263 static bool ExtFlash_readInfo(void)
264 {

287 static bool extFlashVerifyPart(void)
288 {
289  if (!ExtFlash_readInfo())
290  {
291  return false;
292  }
293 
294  pFlashInfo = flashInfo;
295  while (pFlashInfo->deviceSize > 0)
296  {
297  if (infoBuf[0] == pFlashInfo->manfId && infoBuf[1] == pFlashInfo->devId)
298  {
299  break;
300  }
301  pFlashInfo++;
302  }
303 
304  return pFlashInfo->deviceSize > 0;

311 static int ExtFlash_waitReady(void)
312 {
313  const uint8_t wbuf[1] = { BLS_CODE_READ_STATUS };
314  int ret;
315 
316  /* Throw away all garbage */
317  extFlashSelect();
318  Spi_flash();
319  extFlashDeselect();
320 
321  for (;;)
322  {
323  uint8_t buf;
324 
325  extFlashSelect();
326  Spi_write(wbuf, sizeof(wbuf));
327  ret = Spi_read(&buf,sizeof(buf));
328  extFlashDeselect();
329 
330  if (ret)
331  {
332  /* Error */
333  return -2;
334  }
335  if (!(buf & BLS_STATUS_BIT_BUSY))
336  {
337  /* Now ready */
338  break;
339  }
340  }
341 
342  return 0;
343 }

387 bool ExtFlash_open(void)
388 {
389  bool f;
390 
391  hFlashPin = PIN_open(&pinState, BoardFlashPinTable);
392 
393  if (hFlashPin == NULL)
394  {
395  return false;
396  }
397 
398  /* Make sure SPI is available */
399  f = Spi_open(SPI_BIT_RATE);
400 
401  if (f)
402  {
403  /* Put the part is standby mode */
404  f = extFlashPowerStandby();
405 
406  if (f)
407  {
408  /* Verify manufacturer and device ID */
409  f = extFlashVerifyPart();
410  }
411 
412  if (!f)
413  {
414  ExtFlash_close();
415  }
416  }
417 
418  return f;
419 }

441 /* See ExtFlash.h file for description */
442 bool ExtFlash_read(size_t offset, size_t length, uint8_t *buf)
443 {
444  uint8_t wbuf[4];
445 
446  /* Wait till previous erase/program operation completes */
447  int ret = ExtFlash_waitReady();
448  if (ret)
449  {
450  return false;
451  }
452 
453  /* SPI is driven with very low frequency (1MHz < 33MHz fR spec)
454  * in this temporary implementation.
455  * and hence it is not necessary to use fast read. */
456  wbuf[0] = BLS_CODE_READ;
457  wbuf[1] = (offset >> 16) & 0xff;
458  wbuf[2] = (offset >> 8) & 0xff;
459  wbuf[3] = offset & 0xff;
460 
461  extFlashSelect();
462 
463  if (Spi_write(wbuf, sizeof(wbuf)))
464  {
465  /* failure */
466  extFlashDeselect();
467  return false;
468  }
469 
470  ret = Spi_read(buf, length);
471 
472  extFlashDeselect();
473 
474  return ret == 0;
475 }

477 /* See ExtFlash.h file for description */
478 bool ExtFlash_write(size_t offset, size_t length, const uint8_t *buf)
479 {
480  uint8_t wbuf[4];
481 
482  while (length > 0)
483  {
484  /* Wait till previous erase/program operation completes */
485  int ret = ExtFlash_waitReady();
486  if (ret)
487  {
488  return false;
489  }
490 
491  ret = ExtFlash_writeEnable();
492  if (ret)
493  {
494  return false;
495  }
496 
497  size_t ilen; /* interim length per instruction */
498 
499  ilen = BLS_PROGRAM_PAGE_SIZE - (offset % BLS_PROGRAM_PAGE_SIZE);
500  if (length < ilen)
501  {
502  ilen = length;
503  }
504 
505  wbuf[0] = BLS_CODE_PROGRAM;
506  wbuf[1] = (offset >> 16) & 0xff;
507  wbuf[2] = (offset >> 8) & 0xff;
508  wbuf[3] = offset & 0xff;
509 
510  offset += ilen;
511  length -= ilen;
512 
513  /* Up to 100ns CS hold time (which is not clear
514  * whether it's application only in between reads)
515  * is not imposed here since above instructions
516  * should be enough to delay
517  * as much. */
518  extFlashSelect();
519 
520  if (Spi_write(wbuf, sizeof(wbuf)))
521  {
522  /* failure */
523  extFlashDeselect();
524  return false;
525  }
526 
527  if (Spi_write(buf,ilen))
528  {
529  /* failure */
530  extFlashDeselect();
531  return false;
532  }
533  buf += ilen;
534  extFlashDeselect();
535  }
536 
537  return true;
538 }

540 /* See ExtFlash.h file for description */
541 bool ExtFlash_erase(size_t offset, size_t length)
542 {
543  /* Note that Block erase might be more efficient when the floor map
544  * is well planned for OTA but to simplify for the temporary implementation,
545  * sector erase is used blindly. */
546  uint8_t wbuf[4];
547  size_t i, numsectors;
548 
549  wbuf[0] = BLS_CODE_SECTOR_ERASE;
550 
551  {
552  size_t endoffset = offset + length - 1;
553  offset = (offset / BLS_ERASE_SECTOR_SIZE) * BLS_ERASE_SECTOR_SIZE;
554  numsectors = (endoffset - offset + BLS_ERASE_SECTOR_SIZE - 1) /
555  BLS_ERASE_SECTOR_SIZE;
556  }
557 
558  for (i = 0; i < numsectors; i++)
559  {
560  /* Wait till previous erase/program operation completes */
561  int ret = ExtFlash_waitReady();
562  if (ret)
563  {
564  return false;
565  }
566 
567  ret = ExtFlash_writeEnable();
568  if (ret)
569  {
570  return false;
571  }
572 
573  wbuf[1] = (offset >> 16) & 0xff;
574  wbuf[2] = (offset >> 8) & 0xff;
575  wbuf[3] = offset & 0xff;
576 
577  extFlashSelect();
578 
579  if (Spi_write(wbuf, sizeof(wbuf)))
580  {
581  /* failure */
582  extFlashDeselect();
583  return false;
584  }
585  extFlashDeselect();
586 
587  offset += BLS_ERASE_SECTOR_SIZE;
588  }
589 
590  return true;
591 }
