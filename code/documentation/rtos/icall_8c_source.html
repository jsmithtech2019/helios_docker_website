741 static void ICall_msgEnqueue( ICall_MsgQueue *q_ptr, void *msg_ptr )
742 {
743  void *list;
744  ICall_CSState key;
745 
746  // Hold off interrupts
747  key = ICall_enterCSImpl();
748 
749  ICALL_MSG_NEXT( msg_ptr ) = NULL;
750  // If first message in queue
751  if ( *q_ptr == NULL )
752  {
753  *q_ptr = msg_ptr;
754  }
755  else
756  {
757  // Find end of queue
758  for ( list = *q_ptr; ICALL_MSG_NEXT( list ) != NULL;
759  list = ICALL_MSG_NEXT( list ) );
760 
761  // Add message to end of queue
762  ICALL_MSG_NEXT( list ) = msg_ptr;
763  }
764 
765  // Re-enable interrupts
766  ICall_leaveCSImpl(key);
767 }

787  ICALL_MSG_NEXT( msg_ptr ) = NULL;
788  ICALL_MSG_DEST_ID( msg_ptr ) = ICALL_UNDEF_DEST_ID;
789  }
790 
791  // Re-enable interrupts
792  ICall_leaveCSImpl(key);

1467  * Note that though theoretically, Swi or lower priority Hwi may block
