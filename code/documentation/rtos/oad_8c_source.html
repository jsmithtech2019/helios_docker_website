330 /*********************************************************************
331  * @fn OAD_addService
332  *
333  * @brief Initializes the OAD Service by registering GATT attributes
334  * with the GATT server. Only call this function once.
335  *
336  * @param None.
337  *
338  * @return The return value of GATTServApp_RegisterForMsg().
339  */
340 bStatus_t OAD_addService(void)
341 {
342  // Allocate Client Characteristic Configuration table.
343  oadImgIdentifyConfig = (gattCharCfg_t *)ICall_malloc(sizeof(gattCharCfg_t) *
344  linkDBNumConns);
345  if (oadImgIdentifyConfig == NULL)
346  {
347  return bleMemAllocError;
348  }
349 
350  // Allocate Client Characteristic Configuration table.
351  oadImgBlockConfig = (gattCharCfg_t *)ICall_malloc(sizeof(gattCharCfg_t) *
352  linkDBNumConns);
353 
354  if (oadImgBlockConfig == NULL)
355  {
356  // Free already allocated data.
357  ICall_free(oadImgIdentifyConfig);
358 
359  return (bleMemAllocError);
360  }
361 
362  // Allocate Client Characteristic Configuration table.
363  oadStatusConfig = (gattCharCfg_t *)ICall_malloc(sizeof(gattCharCfg_t) *
364  linkDBNumConns);
365 
366  if (oadStatusConfig == NULL)
367  {
368  // Free already allocated data.
369  ICall_free(oadImgIdentifyConfig);
370  ICall_free(oadImgBlockConfig);
371 
372  return (bleMemAllocError);
373  }
374 
375  // Initialize Client Characteristic Configuration attributes.
376  GATTServApp_InitCharCfg(INVALID_CONNHANDLE, oadImgIdentifyConfig);
377  GATTServApp_InitCharCfg(INVALID_CONNHANDLE, oadImgBlockConfig);
378  GATTServApp_InitCharCfg(INVALID_CONNHANDLE, oadStatusConfig);
379 
380  return GATTServApp_RegisterService(oadAttrTbl, GATT_NUM_ATTRS(oadAttrTbl),
381  GATT_MAX_ENCRYPT_KEY_SIZE, &oadCBs);
382 }

399 /*********************************************************************
400  * @fn oadReadAttrCB
401  *
402  * @brief Read an attribute.
403  *
404  * @param connHandle - connection message was received on
405  * @param pAttr - pointer to attribute
406  * @param pValue - pointer to data to be read
407  * @param pLen - length of data to be read
408  * @param offset - offset of the first octet to be read
409  * @param maxLen - maximum length of data to be read
410  * @param method - type of read message
411  *
412  * @return Reads not supported, returns ATT_ERR_INVALID_HANDLE
413  */
414 static bStatus_t oadReadAttrCB(uint16_t connHandle, gattAttribute_t *pAttr,
415  uint8_t *pValue, uint16_t *pLen, uint16_t offset,
416  uint16_t maxLen, uint8_t method)
417 {
418  bStatus_t status;
419 
420  // Make sure it's not a blob operation (no attributes in the profile are long)
421  if (offset > 0)
422  {
423  return ATT_ERR_ATTR_NOT_LONG;
424  }
425 
426  // 128-bit UUID
427  if (!memcmp(pAttr->type.uuid, oadCharUUID[OAD_IDX_IMG_STATUS],
428  ATT_UUID_SIZE))
429  {
430  *pLen = 1;
431  pValue[0] = *pAttr->pValue;
432  status = SUCCESS;
433  }
434  else
435  {
436  *pLen = 0;
437  status = ATT_ERR_INVALID_HANDLE;
438  }
439 
440  return status;
441 }

443 /*********************************************************************
444  * @fn oadWriteAttrCB
445  *
446  * @brief Validate and Write attribute data
447  *
448  * @param connHandle - connection message was received on
449  * @param pAttr - pointer to attribute
450  * @param pValue - pointer to data to be written
451  * @param len - length of data
452  * @param offset - offset of the first octet to be written
453  * @param method - type of write message
454  *
455  * @return SUCCESS, blePending or Failure
456  */
457 static bStatus_t oadWriteAttrCB(uint16_t connHandle, gattAttribute_t *pAttr,
458  uint8_t *pValue, uint16_t len, uint16_t offset,
459  uint8_t method)
460 {
461  bStatus_t status = SUCCESS;
462 
463  if (pAttr->type.len == ATT_BT_UUID_SIZE)
464  {
465  // 16-bit UUID
466  uint16_t uuid = BUILD_UINT16(pAttr->type.uuid[0], pAttr->type.uuid[1]);
467  if (uuid == GATT_CLIENT_CHAR_CFG_UUID)
468  {
469  // Process a CCC write request.
470  status = GATTServApp_ProcessCCCWriteReq(connHandle, pAttr, pValue, len,
471  offset, GATT_CLIENT_CFG_NOTIFY);
472  }
473  else
474  {
475  status = ATT_ERR_ATTR_NOT_FOUND; // Should never get here!
476  }
477  }
478  else
479  {
480  // 128-bit UUID
481  if (!memcmp(pAttr->type.uuid, oadCharUUID[OAD_IDX_IMG_IDENTIFY],
482  ATT_UUID_SIZE))
483  {
484  /* OAD manager has sent new image information.
485  * Read and decide whether to accept or reject an OAD
486  * of this new application image.
487  */
488 
489  // Notify Application
490  if (oadTargetWriteCB != NULL)
491  {
492  oad_imageIdLen = len;
493  (*oadTargetWriteCB)(OAD_WRITE_IDENTIFY_REQ, connHandle, pValue);
494  }
495  }
496  else if (!memcmp(pAttr->type.uuid, oadCharUUID[OAD_IDX_IMG_BLOCK],
497  ATT_UUID_SIZE))
498  {
499  /* OAD is ongoing.
500  * the OAD manager has sent a block from the new image.
501  */
502 
503  // Notify the application.
504  if (oadTargetWriteCB != NULL)
505  {
506  (*oadTargetWriteCB)(OAD_WRITE_BLOCK_REQ, connHandle, pValue);
507  }
508  }
509  else if (!memcmp(pAttr->type.uuid, oadCharUUID[OAD_IDX_IMG_COUNT],
510  ATT_UUID_SIZE))
511  {
512 #ifndef FEATURE_OAD_ONCHIP
513  /*
514  * This is a manipulation of the number of OAD Images to download before
515  * issuing a reset.
516  */
517  if (len == sizeof(uint8_t) && *pValue != 0x00)
518  {
519  oadCharVals[OAD_IDX_IMG_COUNT] = *pValue;
520  }
521  else
522  {
523  status = ATT_ERR_INVALID_VALUE;
524  }
525 #endif // !FEATURE_OAD_ONCHIP
526  }
527  else
528  {
529  status = ATT_ERR_ATTR_NOT_FOUND; // Should never get here!
530  }
531  }
532 
533  return status;
534 }

536 /*********************************************************************
537  * @fn OAD_imgIdentifyWrite
538  *
539  * @brief Process the Image Identify Write. Determined if the image
540  * header identified here should or should not be downloaded by
541  * this application.
542  *
543  * @param connHandle - connection message was received on
544  * @param pValue - pointer to image header data
545  *
546  * @return none
547  */
548 void OAD_imgIdentifyWrite(uint16_t connHandle, uint8_t *pValue)
549 {
550  img_hdr_t ImgHdr;
551  uint8_t hdrOffset = oad_imageIdLen == 8 ? 0 : 4;
552 
553  // Store the new image's header
554  OADTarget_storeImageHeader(pValue);
555 
556  // Read out running image's header.
557  OADTarget_getCurrentImageHeader(&ImgHdr);
558 
559  // Calculate block total of the new image.
560  oadBlkTot = BUILD_UINT16(pValue[hdrOffset + 2], pValue[hdrOffset + 3]) /
561  (OAD_BLOCK_SIZE / HAL_FLASH_WORD_SIZE);
562  oadBlkNum = 0;
563 #ifndef FEATURE_OAD_ONCHIP
564  flagRecord = 0;
565 #endif
566 
567  /* Requirements to begin OAD:
568  * 1) LSB of image version cannot be the same, this would imply a code overlap
569  * between currently running image and new image.
570  * 2) Total blocks of new image must not exceed maximum blocks supported, else
571  * the new image cannot fit.
572  * 3) Block total must be greater than 0.
573  * 4) Optional: Add additional criteria for initiating OAD here.
574  */
575  if (OADTarget_validateNewImage(pValue + hdrOffset, &ImgHdr, oadBlkTot))
576  {
577  // Determine where image will be stored.
578  imageAddress = OADTarget_imageAddress(pValue+hdrOffset);
579  imagePage = imageAddress / HAL_FLASH_PAGE_SIZE;
580 
581  // Open the target interface
582  if (OADTarget_open())
583  {
584  uint8_t page;
585  uint8_t lastPage = oadBlkTot / OAD_BLOCKS_PER_PAGE;
586 
587  // Set last page to end of OAD image address range.
588  lastPage += imagePage;
589 
590  // Erase required pages
591  for (page = imagePage; page <= lastPage; page++)
592  {
593  OADTarget_eraseFlash(page);
594  }
595 
596  // Image accepted, request block 0.
597  OAD_getNextBlockReq(connHandle, 0);
598  }
599  else
600  {
601  // Opening the flash has failed; report error
602  OAD_sendStatus(connHandle, OAD_FLASH_ERR);
603  }
604  }
605  else
606  {
607  // Image rejected, send header information of currently running image
608  // to OAD manager.
609  OAD_rejectImage(connHandle, &ImgHdr);
610  }
611 }

613 /*********************************************************************
614  * @fn OAD_imgBlockWrite
615  *
616  * @brief Process the Image Block Write.
617  *
618  * @param connHandle - connection message was received on
619  * @param pValue - pointer to data to be written
620  *
621  * @return none
622  */
623 void OAD_imgBlockWrite(uint16_t connHandle, uint8_t *pValue)
624 {
625  // N.B. This must be left volatile.
626  volatile uint16_t blkNum = BUILD_UINT16(pValue[0], pValue[1]);
627 
628  // Check that this is the expected block number.
629  if (oadBlkNum == blkNum)
630  {
631  // Write a 16 byte block to Flash.
632  OADTarget_writeFlash(imagePage, (blkNum * OAD_BLOCK_SIZE), pValue+2,
633  OAD_BLOCK_SIZE);
634 
635  // Increment received block count.
636  oadBlkNum++;
637  }
638  else
639  {
640  // Overflow, abort OAD
641  oadBlkNum = 0;
642 #ifndef FEATURE_OAD_ONCHIP
643  flagRecord = 0;
644 #endif
645  // Close the target device
646  OADTarget_close();
647 
648  // Send status
649  OAD_sendStatus(connHandle, OAD_BUFFER_OFL);
650 
651  return;
652  }
653 
654  // Check if the OAD Image is complete.
655  if (oadBlkNum == oadBlkTot)
656  {
657 #if FEATURE_OAD_ONCHIP
658  // Indicate a successful download
659  OAD_sendStatus(connHandle, OAD_SUCCESS);
660 
661  // Notify the application, application handles restart
662  if (oadTargetWriteCB != NULL)
663  {
664  (*oadTargetWriteCB)(OAD_IMAGE_COMPLETE, connHandle, NULL);
665  }
666 #else // !FEATURE_OAD_ONCHIP
667  // Run CRC check on new image.
668  if (checkDL())
669  {
670  // Indicate a successful download and CRC just before rebooting
671  OAD_sendStatus(connHandle, OAD_SUCCESS);
672 
673  // Store the flag of the downloaded image.
674  flagRecord |= getImageFlag();
675 
676  // Store the image information.
677  saveImageInfo();
678 
679  // Check if all expected images have been downloaded.
680  if (CheckImageDownloadCount())
681  {
682  // If one image is a network processor image, inform the application now
683  // so that it can take action on that image.
684  // Note: this callback is not being sent from the context of an
685  // interrupt. It is ok to take any action here.
686  if (flagRecord & OAD_IMG_NP_FLAG)
687  {
688  (*oadTargetWriteCB)(OAD_IMAGE_COMPLETE, connHandle, NULL);
689  }
690 
691  // If one image is an application or stack image, perform the reset
692  // here.
693  if (flagRecord & (OAD_IMG_APP_FLAG|OAD_IMG_STACK_FLAG))
694  {
695  OADTarget_systemReset();
696  }
697 
698  flagRecord = 0;
699  }
700  }
701  else
702  {
703  // CRC error
704  OAD_sendStatus(connHandle, OAD_CRC_ERR);
705  }
706  flagRecord = 0;
707 #endif //FEATURE_OAD_ONCHIP
708 
709  OADTarget_close();
710  oadBlkNum = 0;
711  }
712  else
713  {
714  // Request the next OAD Image block.
715  OAD_getNextBlockReq(connHandle, oadBlkNum);
716  }
717 }

719 /*********************************************************************
720  * @fn OAD_getNextBlockReq
721  *
722  * @brief Process the Request for next image block.
723  *
724  * @param connHandle - connection message was received on
725  * @param blkNum - block number to request from OAD Manager.
726  *
727  * @return None
728  */

859 /*********************************************************************
860  * @fn crcCalcDL
861  *
862  * @brief Run the CRC16 Polynomial calculation over the DL image.
863  *
864  * @param None
865  *
866  * @return The CRC16 calculated.
867  */
868 static uint16_t crcCalcDL(void)
869 {
870  uint16_t imageCRC = 0;
871  uint8_t page;
872  uint8_t lastPage = oadBlkTot / OAD_BLOCKS_PER_PAGE;
873 
874  // Remainder of bytes not divisible by the size of a flash page in bytes.
875  uint16_t numRemBytes = (oadBlkTot - (lastPage * OAD_BLOCKS_PER_PAGE))
876  * OAD_BLOCK_SIZE;
877 
878  // Set last page to end of OAD image address range.
879  lastPage += imagePage;
880 
881  // Read over downloaded pages
882  for (page = imagePage; page <= lastPage; page++)
883  {
884  uint16_t offset;
885 
886  // Read over all flash words in a page, excluding the CRC section of the
887  // first page and all bytes after remainder bytes on the last page.
888  for (offset = (page == imagePage) ? HAL_FLASH_WORD_SIZE : 0;
889  offset < HAL_FLASH_PAGE_SIZE &&
890  (page < lastPage || offset < numRemBytes);
891  offset += HAL_FLASH_WORD_SIZE)
892  {
893  uint8_t buf[HAL_FLASH_WORD_SIZE];
894  uint8_t idx;
895 
896  // Read a word from flash.
897  OADTarget_readFlash(page, offset, buf, HAL_FLASH_WORD_SIZE);
898 
899  // Calculate CRC of word, byte by byte.
900  for (idx = 0; idx < HAL_FLASH_WORD_SIZE; idx++)
901  {
902  imageCRC = crc16(imageCRC, buf[idx]);
903  }
904  }
905  }
906 
907  // IAR note explains that poly must be run with value zero for each byte of
908  // the crc.
909  imageCRC = crc16(imageCRC, 0);
910  imageCRC = crc16(imageCRC, 0);
911 
912  // Return the CRC calculated over the image.
913  return imageCRC;
914 }

916 /*********************************************************************

918  *
919  * @brief Check validity of the downloaded image.
920  *
921  * @param None.
922  *
923  * @return TRUE or FALSE for image valid.
924  */
925 static uint8_t checkDL(void)

942  OADTarget_setCrc(crc);

948 /*********************************************************************
949  * @fn crc16
950  *
951  * @brief Run the CRC16 Polynomial calculation over the byte parameter.
952  *
953  * @param crc - Running CRC calculated so far.
954  * @param val - Value on which to run the CRC16.
955  *
956  * @return crc - Updated for the run.
957  */
958 static uint16_t crc16(uint16_t crc, uint8_t val)
959 {
960  const uint16_t poly = 0x1021;
961  uint8_t cnt;
962 
963  for (cnt = 0; cnt < 8; cnt++, val <<= 1)
964  {
965  uint8_t msb = (crc & 0x8000) ? 1 : 0;
966 
967  crc <<= 1;
968 
969  if (val & 0x80)
970  {
971  crc |= 0x0001;
972  }
973 
974  if (msb)
975  {
976  crc ^= poly;
977  }
978  }
979 
980  return crc;
981 }
