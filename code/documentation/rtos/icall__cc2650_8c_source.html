92 static int_least32_t
93 ICallPlatform_pwrNotify(unsigned int eventType, uintptr_t eventArg,
94  uintptr_t clientArg)
95 {
96  ICall_PwrNotifyData *data = (ICall_PwrNotifyData *) clientArg;
97  ICall_PwrTransition transition;
98  switch (eventType)
99  {
100  case PowerCC26XX_AWAKE_STANDBY:
101  transition = ICALL_PWR_AWAKE_FROM_STANDBY;
102  break;
103  case PowerCC26XX_ENTERING_STANDBY:
104  transition = ICALL_PWR_ENTER_STANDBY;
105  break;
106  case PowerCC26XX_ENTERING_SHUTDOWN:
107  transition = ICALL_PWR_ENTER_SHUTDOWN;
108  break;
109  case PowerCC26XX_AWAKE_STANDBY_LATE:
110  transition = ICALL_PWR_AWAKE_FROM_STANDBY_LATE;
111  break;
112  default:
113  ICALL_HOOK_ABORT_FUNC();
114  break;
115  }
116  data->_private(transition, data);
117  return Power_NOTIFYDONE;
118 }

126 static void ICallPlatform_pwrRequireImpl(uint_fast32_t bitmap)
127 {
128  for (;;)
129  {
130 #ifdef __IAR_SYSTEMS_ICC__
131  char pos = __CLZ(bitmap);
132 #else
133  int pos = __clz(bitmap);
134 #endif
135  if (pos == 32)
136  break;
137  pos = 31 - pos;
138  bitmap ^= 1ul << pos;
139  if (pos < ICALLCC2650_PWR_CFG_D_OFFSET)
140  {
141  Power_setConstraint((uint_least32_t) (pos));
142  }
143  else
144  {
145  Power_setDependency(pos - ICALLCC2650_PWR_CFG_D_OFFSET);
146  }
147  }
148 }

181 ICall_Errno
182 ICallPlatform_pwrUpdActivityCounter(ICall_PwrUpdActivityCounterArgs *args)
183 {
184  args->pwrRequired = TRUE;
185  if (args->incFlag)
186  {
187  /* Increment counter */
188  ICall_CSState state = ICall_enterCSImpl();
189  if (ICallPlatform_pwrActivityCount == 255)
190  {
191  /* Counter value overflow */
192  ICall_leaveCSImpl(state);
193  return ICALL_ERRNO_OVERFLOW;
194  }
195  ICallPlatform_pwrActivityCount++;
196  if (ICallPlatform_pwrActivityCount == 1)
197  {
198  /* Needs to stay in power active state.
199  * Set constraints and dependencies as configured. */
200  ICallPlatform_pwrRequireImpl(ICallPlatform_pwrCfgACAction);
201  }
202  ICall_leaveCSImpl(state);
203  }
204  else
205  {
206  /* Decrement counter */
207  ICall_CSState state = ICall_enterCSImpl();
208  if (ICallPlatform_pwrActivityCount == 0)
209  {
210  /* Counter value underflow */
211  ICall_leaveCSImpl(state);
212  return ICALL_ERRNO_UNDERFLOW;
213  }
214  ICallPlatform_pwrActivityCount--;
215  if (ICallPlatform_pwrActivityCount == 0)
216  {
217  /* No activity.
218  * Lift the constraints and dependencies as configured. */
219  ICall_leaveCSImpl(state);
220  ICallPlatform_pwrDispenseImpl(ICallPlatform_pwrCfgACAction);
221  args->pwrRequired = FALSE;
222  }
223  else
224  {
225  ICall_leaveCSImpl(state);
226  }
227  }
228  return ICALL_ERRNO_SUCCESS;
229 }

233 ICallPlatform_pwrConfigACAction(ICall_PwrBitmapArgs *args)
234 {
235  ICall_CSState state;
236 
237  if (args->bitmap & ICALLCC2650_PWR_CFG_NMASK)
238  {
239  return ICALL_ERRNO_INVALID_PARAMETER;
240  }
241 
242  state = ICall_enterCSImpl();
243 
244  if (ICallPlatform_pwrActivityCount > 0)
245  {
246  /* Release previous constraints */
247  ICallPlatform_pwrDispenseImpl(ICallPlatform_pwrCfgACAction);
248  /* Set new constraints */
249  ICallPlatform_pwrRequireImpl(args->bitmap);
250  }
251  ICallPlatform_pwrCfgACAction = args->bitmap;
252 
253  ICall_leaveCSImpl(state);
254  return ICALL_ERRNO_SUCCESS;
255 }

411 void ICallPlatform_verify(void)
412 {
413  size_t i;
414  for (i = 0; i < sizeof(ICallPlatform_pwrConstraintMap) /
415  sizeof(ICallPlatform_pwrConstraintMap[0]); i++)
416  {
417  if (ICallPlatform_pwrConstraintMap[i].icallid !=
418  ICallPlatform_pwrConstraintMap[i].pwrid)
419  {
420  ICall_abort();
421  }
422  if (ICallPlatform_pwrConstraintMap[i].icallid &
423  ICALLCC2650_PWR_CFG_NMASK)
424  {
425  ICall_abort();
426  }
427  }
428  for (i = 0; i < sizeof(ICallPlatform_pwrResourceMap) /
429  sizeof(ICallPlatform_pwrResourceMap[0]); i++)
430  {
431  if (ICallPlatform_pwrResourceMap[i].icallid !=
432  (1ull << (ICALLCC26XX_PWR_CFG_D_OFFSET +
433  ICallPlatform_pwrResourceMap[i].pwrid)))
434  {
435  ICall_abort();
436  }
437  if (ICallPlatform_pwrResourceMap[i].icallid &
438  ICALLCC26XX_PWR_CFG_NMASK)
439  {
440  ICall_abort();
441  }
442  }
443  for (i = 0; i < sizeof(ICallPlatform_pwrTransitionStateMap) /
444  sizeof(ICallPlatform_pwrTransitionStateMap[0]); i++)
445  {
446  if (ICallPlatform_pwrTransitionStateMap[i].icallid !=
447  ICallPlatform_pwrTransitionStateMap[i].pwrid)
448  {
449  ICall_abort();
450  }
451  }
452  if (ICall_pwrGetTransitionState() != ICALL_PWR_TRANSITION_STAY_IN_ACTIVE)
453  {
454  ICall_abort();
455  }
456 }
