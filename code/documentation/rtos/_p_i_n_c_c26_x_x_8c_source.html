90 // I/O SWI service routine that posts the callback in a SWI context
91 static void PIN_swi(UArg arg0, UArg arg1){
92 
93  uint32_t iEvent;
94  unsigned char iEventCounter;
95  PIN_Handle handle;
96  PIN_IntCb pCb;
97 
98  // Get the OR'd trigger value representing all iEvent values prior to running the SWI
99  iEvent = Swi_getTrigger();
100 
101  // iEventCounter cycles through all pins on the device up to the max number of pins
102  for(iEventCounter = 0; iEventCounter < PIN_NumPins; iEventCounter++){
103  // Check if current iEventCounter bit is set in iEvent
104  if(iEvent & (1 << iEventCounter)){
105  // Get pin handle and registered callback function
106  // Double paranthesis to supress GCC warning
107  // Intentional assignment is intended, not the equality comparison
108  if((handle = PIN_HandleTable[iEventCounter])) {
109  if((pCb = handle->pCbFunc)) {
110  // Event from existing pin, with an associated handle and a
111  // registered callback -> call callback
112  // Run the callback function in a SWI context.
113  pCb(handle, iEventCounter);
114  }
115  }
116  }
117  }
118 }

120 // I/O HWI service routine
121 static void PIN_hwi(UArg arg) {
122  uint32_t iEvent;
123 
124  // Get event flag with lowest index (also pin ID)
125  iEvent = HWREG(GPIO_BASE + GPIO_O_EVFLAGS31_0);
126 
127  // Clear this event flag
128  HWREG(GPIO_NONBUF_BASE + GPIO_O_EVFLAGS31_0) = iEvent;
129 
130  // Include all GPIO's currently triggered in the SWI
131  Swi_or(Swi_handle(&(PinSwi)), iEvent);
132 }

134 // Internal utility function for setting IOCFG register for pin
135 static void PINCC26XX_setIoCfg(PIN_Config bmMask, PIN_Config pinCfg) {
136  uint32_t dsCfg;
137  PIN_Id pinId = PIN_ID(pinCfg);
138  bool bInvChanges;
139 
140  if (pinCfg&PIN_GEN) {
141  // Translate from device-independent to device-specific PIN_Config values
142  pinCfg ^= PIN_GEN|PIN_BM_INPUT_EN|PIN_BM_PULLING;
143  }
144  // Get existing IOCFG, determine whether inversion changes, mask away what will be updated
145  dsCfg = HWREG(IOC_BASE+IOC_O_IOCFG0+4*pinId);
146  bInvChanges = (dsCfg^pinCfg)&bmMask&PINCC26XX_INV_INOUT;
147  dsCfg &= ~bmMask;
148 
149  // Insert what we want to update, possibly revert IRQ edges, write back to IOCFG
150  dsCfg |= (pinCfg&bmMask&PINCC26XX_BM_IOCFG);
151  if ((bmMask&PINCC26XX_BM_IRQ)==PINCC26XX_BM_IRQ && (dsCfg&PINCC26XX_INV_INOUT)==0) {
152  // We're changing IRQ options but inversion will not be enabled -> keep IRQ options
153  } else if ((bmMask&PINCC26XX_BM_IRQ)==0 && !bInvChanges) {
154  // We're not changing IRQ options and inversion remains unchanged -> keep IRQ options
155  } else {
156  // We're updating IRQ options and inversion will be enabled, OR
157  // we're not updating IRQ options but inversion settings change
158  // -> reverse polarity of edge detection when positive-only or negative-only
159  switch (dsCfg&PINCC26XX_BM_IRQ) {
160  case PINCC26XX_IRQ_POSEDGE:
161  dsCfg &= ~PINCC26XX_BM_IRQ;
162  dsCfg |= PINCC26XX_IRQ_NEGEDGE;
163  break;
164  case PINCC26XX_IRQ_NEGEDGE:
165  dsCfg &= ~PINCC26XX_BM_IRQ;
166  dsCfg |= PINCC26XX_IRQ_POSEDGE;
167  break;
168  default:
169  break;
170  }
171  }
172  HWREG(IOC_BASE+IOC_O_IOCFG0+4*pinId) = dsCfg;
173 
174  // Update GPIO output value and enable depending on previous output mode (enabled or disabled)
175  {
176  bool bOutputEnabled = (HWREG(GPIO_BASE+GPIO_O_DOE31_0)&(1<<pinId)) ? true : false;
177 
178  if(!bOutputEnabled) {
179  if (bmMask&PINCC26XX_BM_GPIO_OUTPUT_VAL) {
180  // Set GPIO output value
181  HWREGB(GPIO_BASE+GPIO_O_DOUT3_0+pinId) = (pinCfg&PINCC26XX_BM_GPIO_OUTPUT_VAL) ? 1 : 0;
182  }
183  }
184 
185  if (bmMask&PINCC26XX_BM_GPIO_OUTPUT_EN) {
186  // Set GPIO output enable
187  uint32_t key = Hwi_disable();
188  HWREG(GPIO_BASE+GPIO_O_DOE31_0) =
189  (HWREG(GPIO_BASE+GPIO_O_DOE31_0) & ~(1<<pinId)) |
190  ((pinCfg&PINCC26XX_BM_GPIO_OUTPUT_EN) ? (1<<pinId) : 0);
191  Hwi_restore(key);
192  }
193 
194  if(bOutputEnabled) {
195  if (bmMask&PINCC26XX_BM_GPIO_OUTPUT_VAL) {
196  // Set GPIO output value
197  HWREGB(GPIO_BASE+GPIO_O_DOUT3_0+pinId) = (pinCfg&PINCC26XX_BM_GPIO_OUTPUT_VAL) ? 1 : 0;
198  }
199  }
200  }
201 }

205 // Internal utility function for setting mux setting in IOCFG register for pin
206 static void PINCC26XX_setIoCfgMux(PIN_Id pinId, int32_t nMux) {
207  // Read in existing value in IOCFG register and update with supplied mux value
208  if (nMux<0) {
209  nMux = PINCC26XX_MUX_GPIO;
210  }
211  uint32_t dsCfg;
212  dsCfg = HWREG(IOC_BASE+IOC_O_IOCFG0+4*pinId);
213  dsCfg &= ~IOC_IOCFG0_PORT_ID_M;
214  dsCfg |= nMux & IOC_IOCFG0_PORT_ID_M;
215  HWREG(IOC_BASE+IOC_O_IOCFG0+4*pinId) = dsCfg;
216 }

218 uint_t PINCC26XX_getPinCount(){
219  // Get number of pins available on device (from HW register)
220  uint_t pinCount = (( HWREG( FCFG1_BASE + FCFG1_O_IOCONF ) &
221  FCFG1_IOCONF_GPIO_CNT_M ) >>
222  FCFG1_IOCONF_GPIO_CNT_S ) ;
223  // HW register will report 30 DIOs for CC13xx 7x7 (which is correct).
224  // However the PIN driver is full of checks to make sure PIN_ID is not out of range.
225  // Add one for CC13xx, DIO_0 does not exist and has been reserved, so the rest
226  // can safely assume that DIO_1 to DIO_30 (on 7x7 QFN) are available
227  if(ChipInfo_GetChipFamily()==FAMILY_CC13xx && ChipInfo_GetPackageType()==PACKAGE_7x7) {
228  pinCount++;
229  }
230  return pinCount;
231 }

233 PIN_Status PIN_init(const PIN_Config aPinCfg[]) {
234  uint_t i;
235  uint32_t bmInit; // Works as long as # pins <=32
236  Semaphore_Params semParams;
237  Hwi_Params hwiParams;
238  Swi_Params swiParams;
239  uint32_t bmReservedPins = 0;
240 
241  // Its ok if Power init has already been called.
242  Power_init();
243 
244  // Mark pins that aren't available up front
245  if (ChipInfo_GetChipFamily()==FAMILY_CC13xx && ChipInfo_GetPackageType()==PACKAGE_7x7) {
246  // CC13xx 7x7 QFN does not have IO #0
247  bmReservedPins |= (1<<0);
248  }
249  if ((HWREG(CCFG_BASE+CCFG_O_MODE_CONF)&CCFG_MODE_CONF_SCLK_LF_OPTION_M)>>CCFG_MODE_CONF_SCLK_LF_OPTION_S == 0x1) {
250  // An IO is used for LF clock input
251  bmReservedPins |= (1<<((HWREG(CCFG_BASE+CCFG_O_EXT_LF_CLK)&CCFG_EXT_LF_CLK_DIO_M)>>CCFG_EXT_LF_CLK_DIO_S));
252  }
253 
254  // Make sure we are using correct version of Driverlib
255  DRIVERLIB_ASSERT_CURR_RELEASE();
256 
257  // Get number of pins available on device (from HW register)
258  PIN_NumPins = PINCC26XX_getPinCount();
259 
260  // Initialize table of pins that have default GPIO configuration
261  for (i=0; i<PIN_NumPins; i++) {
262  PinGpioConfigTable[i] = PIN_UNASSIGNED;
263  }
264 
265  // Read in aPinCfg list and create bitmask of which IOs to initialize to
266  // default values and which to initialize from aPinConfig
267  for (i=bmInit=0; PIN_ID(aPinCfg[i])!=PIN_TERMINATE; i++) {
268  // Ignore unassigned pins
269  if (PIN_ID(aPinCfg[i]) == PIN_UNASSIGNED) {
270  continue;
271  }
272  // Check that pin exists and is available
273  if (PIN_ID(aPinCfg[i])>=PIN_NumPins || bmReservedPins&(1<<PIN_ID(aPinCfg[i]))) {
274  return PIN_NO_ACCESS;
275  }
276  // Mark pin as being in aPinConfig
277  bmInit |= (1<<PIN_ID(aPinCfg[i]));
278  // For quick reference, store index i in table
279  PinGpioConfigTable[PIN_ID(aPinCfg[i])] = i;
280  }
281 
282  // Set Power dependecies & constraints
283  Power_setDependency(PowerCC26XX_PERIPH_GPIO);
284 
285  // Save GPIO default setup
286  pPinGpio = aPinCfg;
287 
288  // Setup semaphore for sequencing accesses to PIN_open()
289  Semaphore_Params_init(&semParams);
290  semParams.mode = Semaphore_Mode_BINARY;
291  Semaphore_construct(&PinSem, 1, &semParams);
292 
293  // Loop thru all pins and configure
294  for (i=0; i<PIN_NumPins; i++) {
295  if (bmReservedPins&(1<<i)) {
296  // Pin is reserved for other purposes -> setup dummy handle
297  PIN_HandleTable[i] = (PIN_State*)0x00000004;
298  } else {
299  if (bmInit&(1<<i)) {
300  // Setup all pins in aPinCfg as instructed
301  PINCC26XX_setIoCfg(PIN_BM_ALL, aPinCfg[PinGpioConfigTable[i]]);
302  } else {
303  // Setup all pins not in aPinCfg to default configuration:
304  // GPIO, input buffer disable, GPIO output disable, low GPIO output, no pull, no IRQ, no wakeup
305  PINCC26XX_setIoCfg(PIN_BM_ALL, PIN_ID(i)|PINCC26XX_NOPULL);
306  }
307  // Set pin as GPIO and clear pin handle
308  PINCC26XX_setIoCfgMux(PIN_ID(i), -1);
309  PIN_HandleTable[i] = NULL;
310  }
311  }
312 
313  // Setup HWI handler
314  Hwi_Params_init(&hwiParams);
315  hwiParams.priority = PINCC26XX_hwAttrs.intPriority;
316  Hwi_construct(&PinHwi, INT_AON_GPIO_EDGE, PIN_hwi,&hwiParams, NULL);
317 
318  // Setup SWI handler
319  Swi_Params_init(&(swiParams));
320  swiParams.priority = PINCC26XX_hwAttrs.swiPriority;
321  swiParams.trigger = 0;
322  Swi_construct(&PinSwi, PIN_swi, &(swiParams), NULL);
323 
324  // Setup interrupts so that they wake up from standby (use MCU_WU1)
325  HWREG(AON_EVENT_BASE+AON_EVENT_O_MCUWUSEL) =
326  (HWREG(AON_EVENT_BASE+AON_EVENT_O_MCUWUSEL)&(~AON_EVENT_MCUWUSEL_WU1_EV_M)) |
327  AON_EVENT_MCUWUSEL_WU1_EV_PAD;
328 
329  // Open latches out to I/Os
330  // This might be unnecessary, but isn't when you start from debugger
331  HWREG(AON_IOC_BASE+AON_IOC_O_IOCLATCH) = AON_IOC_IOCLATCH_EN;
332 
333  // If we boot from shutdown, the IOs are latched, this opens the latches again
334  HWREG(AON_SYSCTL_BASE+AON_SYSCTL_O_SLEEPCTL) = AON_SYSCTL_SLEEPCTL_IO_PAD_SLEEP_DIS;
335 
336  return PIN_SUCCESS;
337 }

341 PIN_Handle PIN_open(PIN_State* pState, const PIN_Config aPinList[]) {
342  uint_t i;
343  bool bOk;
344  uint32_t bmPort;
345  PIN_Id pinId;
346 
347  // Ensure that only one client at a time can call PIN_open() or PIN_add()
348  Semaphore_pend(Semaphore_handle(&PinSem), BIOS_WAIT_FOREVER);
349 
350  // Check whether all pins in aPinList are valid and available first
351  for (i=0, bOk=true, bmPort=0; aPinList && (pinId=PIN_ID(aPinList[i]))!=PIN_TERMINATE; i++) {
352  /* Unassigned pins is allowed, but cannot generate a bitmask. */
353  if (pinId != PIN_UNASSIGNED) {
354  if (pinId>=PIN_NumPins || PIN_HandleTable[pinId]) {
355  bOk = false;
356  break;
357  } else {
358  // Generate bitmask for port operations (always one port on CC26xx)
359  bmPort |= (1<<pinId);
360  }
361  }
362  }
363 
364  if (!bOk) {
365  // Indicate that the pins were not allocatable
366  pState = NULL;
367  } else {
368  // Setup state object
369  pState->pCbFunc = NULL;
370  pState->bmPort = 0;
371  pState->userArg = 0;
372 
373  // Configure I/O pins according to aPinList
374  for (i=0; aPinList && (pinId=PIN_ID(aPinList[i]))!=PIN_TERMINATE; i++) {
375  if (pinId != PIN_UNASSIGNED) {
376  PIN_HandleTable[pinId] = pState;
377  pState->bmPort |= (1<<pinId);
378  PIN_setConfig(pState, PIN_BM_ALL, aPinList[i]);
379  }
380  }
381  }
382 
383  Semaphore_post(Semaphore_handle(&PinSem));
384  return pState;
385 }

389 PIN_Status PIN_add(PIN_Handle handle, PIN_Config pinCfg) {
390  PIN_Status ret;
391  PIN_Id pinId = PIN_ID(pinCfg);
392 
393  // Check that handle and pinId is valid
394  if (!handle || pinId>=PIN_NumPins) {
395  return PIN_NO_ACCESS;
396  }
397 
398  // Ensure that only one client at a time can call PIN_open() or PIN_add()
399  Semaphore_pend(Semaphore_handle(&PinSem), BIOS_WAIT_FOREVER);
400 
401  // Check whether pin is available
402  if (PIN_HandleTable[pinId]) {
403  // Pin already allocated -> do nothing
404  ret = PIN_ALREADY_ALLOCATED;
405  } else {
406  // Allocate pin
407  PIN_HandleTable[pinId] = handle;
408  handle->bmPort |= (1<<pinId);
409  PIN_setConfig(handle, PIN_BM_ALL, pinCfg);
410  ret = PIN_SUCCESS;
411  }
412 
413  Semaphore_post(Semaphore_handle(&PinSem));
414  return ret;
415 }

419 PIN_Status PIN_remove(PIN_Handle handle, PIN_Id pinId) {
420  if (handle && (handle->bmPort&(1<<pinId))) {
421  // Deallocate pin
422  handle->bmPort &= ~(1<<pinId);
423  PIN_HandleTable[pinId] = NULL;
424  // Find GPIO default value and revert to it
425  if (PinGpioConfigTable[pinId]==PIN_UNASSIGNED) {
426  // Revert pin to default configuration:
427  // GPIO, input buffer disable, GPIO output disable, low GPIO output, no pull, no IRQ, no wakeup
428  PINCC26XX_setIoCfg(PIN_BM_ALL, PIN_ID(pinId)|PIN_INPUT_DIS);
429  } else {
430  // Revert pin to previous GPIO configuration
431  PINCC26XX_setIoCfg(PIN_BM_ALL, pPinGpio[PinGpioConfigTable[pinId]]);
432  }
433  // Revert to GPIO
434  PINCC26XX_setIoCfgMux(PIN_ID(pinId), -1);
435  return PIN_SUCCESS;
436  } else {
437  return PIN_NO_ACCESS;
438  }
439 }

443 void PIN_close(PIN_Handle handle) {
444  uint_t i;
445 
446  // No need for sequencing accesses to PIN_close()
447  // For each pin in port bitmask
448  while (handle->bmPort) {
449  // Find lowest index pin
450  i = PIN_ctz(handle->bmPort);
451  // Deallocate pin
452  PIN_remove(handle, i);
453  }
454 }

543 PIN_Config PINCC26XX_getConfig(PIN_Id pinId) {
544  // Get IOCFG register value and add in some extras:
545  // * pinId
546  // * pin GPIO output enable
547  // * pin GPIO output value
548  uint32_t dsCfg;
549 
550  dsCfg = HWREG(IOC_BASE+IOC_O_IOCFG0+4*pinId);
551  dsCfg &= PINCC26XX_BM_IOCFG;
552  dsCfg |= PIN_ID(pinId);
553  dsCfg |= (PINCC26XX_getOutputValue(pinId)) ? PINCC26XX_GPIO_HIGH : PINCC26XX_GPIO_LOW;
554  dsCfg |= (HWREG(GPIO_BASE+GPIO_O_DOE31_0)&(1<<pinId)) ? PINCC26XX_GPIO_OUTPUT_EN : 0;
555  return dsCfg;
556 }
