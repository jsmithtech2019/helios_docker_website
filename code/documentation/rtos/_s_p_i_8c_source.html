81  /* Call each driver's init function */
82  for (SPI_count = 0; SPI_config[SPI_count].fxnTablePtr != NULL; SPI_count++) {
83  SPI_config[SPI_count].fxnTablePtr->initFxn((SPI_Handle)&(SPI_config[SPI_count]));
84  }
85  }
86 }

89  * ======== SPI_open ========
90  */
91 SPI_Handle SPI_open(unsigned int index, SPI_Params *params)
92 {
93  SPI_Handle handle;
94 
95  if (index >= SPI_count) {
96  return (NULL);
97  }
98 
99  /* If params are NULL use defaults */
100  if (params == NULL) {
101  params = (SPI_Params *) &SPI_defaultParams;
102  }
103 
104  /* Get handle for this driver instance */
105  handle = (SPI_Handle)&(SPI_config[index]);
106 
107  return (handle->fxnTablePtr->openFxn(handle, params));
108 }
