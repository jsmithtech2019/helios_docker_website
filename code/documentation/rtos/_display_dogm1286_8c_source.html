98 Display_Handle DisplayDogm1286_open(Display_Handle hDisplay,
99  Display_Params *params)
100 {
101  DisplayDogm1286_HWAttrs *hwAttrs = (DisplayDogm1286_HWAttrs *)hDisplay->hwAttrs;
102  DisplayDogm1286_Object *object = (DisplayDogm1286_Object *)hDisplay->object;
103 
104  PIN_Config pinTable[1 + 1];
105 
106  uint32_t i = 0;
107  if (hwAttrs->powerPin != PIN_TERMINATE)
108  {
109  pinTable[i++] = hwAttrs->powerPin | PIN_GPIO_OUTPUT_EN | PIN_GPIO_HIGH | PIN_PUSHPULL | PIN_DRVSTR_MAX;
110  }
111  pinTable[i++] = PIN_TERMINATE;
112 
113  object->hPins = PIN_open(&object->pinState, pinTable);
114  if (object->hPins == NULL)
115  {
116  Log_error0("Couldn't open pins for DOGM1286");
117  return NULL;
118  }
119 
120  object->lineClearMode = params->lineClearMode;
121 
122  object->lcdBuffers[0].pcBuffer = object->lcdBuffer0;
123  object->lcdBuffers[0].bufSize = 1024;
124  object->lcdBuffers[0].bufMutex = object->lcdMutex;
125 
126  LCD_Params lcdParams;
127  LCD_Params_init(&lcdParams);
128  object->hLcd = LCD_open(&object->lcdBuffers[0], 1, &lcdParams);
129 
130  if (object->hLcd)
131  {
132  LCD_bufferClear(object->hLcd, 0);
133  LCD_update(object->hLcd, 0);
134  return hDisplay;
135  }
136  else
137  {
138  PIN_close(object->hPins);
139  return NULL;
140  }
141 }

211 void DisplayDogm1286_put5(Display_Handle hDisplay, uint8_t line,
212  uint8_t column, uintptr_t fmt, uintptr_t a0,
213  uintptr_t a1, uintptr_t a2, uintptr_t a3, uintptr_t a4)
214 {
215  DisplayDogm1286_Object *object = (DisplayDogm1286_Object *)hDisplay->object;
216 
217  char dispStr[22] = { 0 };
218  uint8_t xp, clearStartX, clearEndX;
219 
220  xp = column * 6;
221  clearStartX = clearEndX = xp;
222 
223  switch (object->lineClearMode)
224  {
225  case DISPLAY_CLEAR_LEFT:
226  clearStartX = 0;
227  break;
228  case DISPLAY_CLEAR_RIGHT:
229  clearEndX = 127;
230  break;
231  case DISPLAY_CLEAR_BOTH:
232  clearStartX = 0;
233  clearEndX = 127;
234  break;
235  case DISPLAY_CLEAR_NONE:
236  default:
237  break;
238  }
239 
240  if (clearStartX != clearEndX)
241  {
242  LCD_bufferClearPart(object->hLcd, 0,
243  clearStartX, clearEndX, (LCD_Page)line, (LCD_Page)(line));
244  }
245 
246 
247  System_snprintf(dispStr, sizeof(dispStr), (xdc_CString)fmt, a0, a1, a2, a3, a4);
248 
249  LCD_bufferPrintString(object->hLcd, 0, dispStr, xp, (LCD_Page)line);
250  LCD_update(object->hLcd, 0);
251 }
