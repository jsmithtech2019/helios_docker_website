1 /******************************************************************************
2  @file: ble_user_config.h
3 
4  @brief: This file contains user configurable variables for the BLE
5  Controller and Host. This file is also used for defining the
6  type of RF Front End used with the TI device, such as using
7  a Differential front end with External Bias, etc. Please see
8  below for more detail.
9 
10  To change the default values of configurable variables:
11  - Include the followings in your application main.c file:
12  #ifndef USE_DEFAULT_USER_CFG
13 
14  #include "ble_user_config.h"
15 
16  // BLE user defined configuration
17  bleUserCfg_t user0Cfg = BLE_USER_CFG;
18 
19  #endif // USE_DEFAULT_USER_CFG
20  - Set the preprocessor symbol MAX_NUM_BLE_CONNS, MAX_NUM_PDU,
21  MAX_PDU_SIZE, L2CAP_NUM_PSM or L2CAP_NUM_CO_CHANNELS
22  to a desired value in your application project.
23  - Include "ble_user_config.h" in your stack OSAL_ICallBle.c
24  file.
25  - Call setBleUserConfig at the start of stack_main. Actually,
26  it is okay to set the variables anywhere in stack_main as
27  long as it is BEFORE osal_init_system, but best to set at
28  the very start of stack_main.
29 
30  Note: User configurable variables are only used during the
31  initialization of the Controller and Host. Changing
32  the values of these variables after this will have no
33  effect.
34 
35  Note: To use the default user configurable variables, define
36  the preprocessor symbol USE_DEFAULT_USER_CFG in your
37  application project.
38 
39  For example:
40  - In your application main.c, include:
41  #ifndef USE_DEFAULT_USER_CFG
42 
43  #include "bleUserConfig.h"
44 
45  // BLE user defined configuration
46  bleUserCfg_t user0Cfg = BLE_USER_CFG;
47  #endif // USE_DEFAULT_USER_CFG
48  - In your application project, set the preprocessor symbol
49  MAX_NUM_BLE_CONNS to 1 to change the maximum number of BLE
50  connections to 1 from the default value of 3.
51  - In your stack OSAL_ICallBle.c file, call setBleUserCfg to
52  update the user configuration variables:
53  #include "bleUserConfig.h"
54  :
55  int stack_main( void *arg )
56  {
57  setBleUserConfig( (bleUserCfg_t *)arg );
58  :
59  }
60

226 //
227 // Device Package and Evaluation Module (EM) Board
228 //
229 // The CC26xx device comes in three types of packages: 7x7, 5x5, 4x4.
230 // For each package, the user may change how the RF Front End (FE) is
231 // configured. The possible FE settings are provided as a set of defines.
232 // (The user can also set the FE bias, the settings of which are also provided
233 // as defines.) The user can change the value of RF_FE_MODE_AND_BIAS to
234 // configure the RF FE as desired. However, while setting the FE configuration
235 // determines how the device is configured at the package, it is the PCB the
236 // device is mounted on (the EM) that determines how those signals are routed.
237 // So while the FE is configurable, how signals are used is fixed by the EM.
238 // As can be seen, the value of RF_FE_MODE_AND_BIAS is organized by the EM
239 // board as defined by EMs produced by Texas Instruments Inc. How the device
240 // is mounted, routed, and configured for a user product would of course be
241 // user defined, and the value of RF_FE_MODE_AND_BIAS would have to be set
242 // accordingly; the user could even dispense with the conditional board
243 // compiles entirely. So too with the usage of the Tx Power tables. As can be
244 // seen in bleUserConfig.c, there are two tables, one for packages using a
245 // differential FE, and one for single-end. This too has been organized by TI
246 // defined EMs and would have to be set appropriately by the user.
247 //
248 // For example:
249 // Let's say you decided to build several boards using the CC26xx 4x4 package.
250 // For one board, you plan to use a differential RF FE, while on the other you
251 // wish to use a single ended RFN RF FE. You would then create your own board.h
252 // (located by your preprocessor Include path name) that contains one of two
253 // defines that you create (say MY_CC26xx_4ID and MY_CC26xx_4XS). Then you can
254 // define your own choice of RF FE conditionally based on these defines
255 // (completely replacing those shown below), as follows:
256 //
257 //#if defined( MY_CC26xx_4ID )
258 //
259 // #define RF_FE_MODE_AND_BIAS ( RF_FE_DIFFERENTIAL | \
260 // RF_FE_INT_BIAS )
261 //#elif defined( MY_CC26xx_4XS )
262 //
263 // #define RF_FE_MODE_AND_BIAS ( RF_FE_SINGLE_ENDED_RFN | \
264 // RF_FE_EXT_BIAS )
265 //#else // unknown device package
266 // :
267 //
268 // In this way, you can define your own board I/O settings, and configure your
269 // RF FE based on your own board defines.
270 //
271 // For additional details and examples, please see the Software Developer's
272 // Guide.
273 //
274 
275 // RF Front End Mode and Bias Configuration
