82 UDMACC26XX_Handle UDMACC26XX_open()
83 {
84  Hwi_Params hwiParams;
85  unsigned int key;
86  UDMACC26XX_Object *object;
87  UDMACC26XX_HWAttrs const *hwAttrs;
88  UDMACC26XX_Handle handle;
89 
90  /* Get the pointer to the object and the hwAttrs */
91  handle = (UDMACC26XX_Handle)&(UDMACC26XX_config[0]);
92  hwAttrs = handle->hwAttrs;
93  object = handle->object;
94 
95  /* Power up and enable clocks for uDMA. */
96  Power_setDependency(hwAttrs->powerMngrId);
97 
98  /* Disable preemption while checking if the UDMACC26XX is open. */
99  key = Hwi_disable();
100 
101  if(!object->isOpen){
102  Hwi_Params_init(&hwiParams);
103  hwiParams.arg = (UArg) handle;
104  hwiParams.priority = hwAttrs->intPriority;
105  Hwi_construct(&(object->hwi), (int) hwAttrs->intNum, UDMACC26XX_hwiIntFxn, &hwiParams, NULL);
106 
107  /* make sure to mark the uDMA as opened */
108  object->isOpen = TRUE;
109 
110  /* initialize the UDMACC26XX hardware */
111  UDMACC26XX_initHw(handle);
112  }
113 
114  Hwi_restore(key);
115  return (handle);
116 }

118 /*
119  * ======== UDMACC26XX_close ========
120  *
121  */
122 void UDMACC26XX_close(UDMACC26XX_Handle handle)
123 {
124  unsigned int key;
125  UDMACC26XX_Object *object;
126  UDMACC26XX_HWAttrs const *hwAttrs;
127 
128  /* Get the pointer to the object and hwAttrs */
129  hwAttrs = handle->hwAttrs;
130  object = handle->object;
131 
132  /* Disable preemption while checking if the UDMACC26XX is open. */
133  key = Hwi_disable();
134 
135  /* Only consider to take anything down if uDMA is initialized. */
136  if (object->isOpen) {
137  if (Power_getDependencyCount(hwAttrs->powerMngrId) == 1) {
138  uDMADisable(hwAttrs->baseAddr);
139  Hwi_destruct(&(object->hwi));
140  object->isOpen = FALSE;
141  }
142  Power_releaseDependency(hwAttrs->powerMngrId);
143  }
144 
145  Hwi_restore(key);
146 }
