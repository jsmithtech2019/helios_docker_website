193 /*
194  * ======== ECCROMCC26XX_init ========
195  */
196 void ECCROMCC26XX_init(void)
197 {
198  static uint8_t isInit = 0;
199  unsigned int key;
200 
201  // Enter critical section.
202  key = Hwi_disable();
203 
204  if (!isInit)
205  {
206  Semaphore_Params semParams;
207 
208  // Only initialize once.
209  isInit = 1;
210 
211  // Setup semaphore for sequencing accesses to ECC
212  Semaphore_Params_init(&semParams);
213  semParams.mode = Semaphore_Mode_BINARY;
214  ECC_semaphore = Semaphore_create(1, &semParams, NULL);
215  }
216 
217  // Exit critical section.
218  Hwi_restore(key);
219 }

250 /*
251  * ======== ECCROMCC26XX_genKeys ========
252  */
253 int8_t ECCROMCC26XX_genKeys(uint8_t *privateKey, uint8_t *publicKeyX,
254  uint8_t *publicKeyY, ECCROMCC26XX_Params *params)
255 {
256  int8_t status;
257  uint8_t *randStrBuf;
258  uint8_t *pubKeyXBuf;
259  uint8_t *pubKeyYBuf;
260 
261  // Check key buffers and params.
262  if (privateKey == NULL || publicKeyX == NULL || publicKeyY == NULL ||
263  params == NULL || params->malloc == NULL || params->free == NULL)
264  {
265  // Store status.
266  if (params)
267  {
268  params->status = ECCROMCC26XX_STATUS_ILLEGAL_PARAM;
269  }
270 
271  return ECCROMCC26XX_STATUS_ILLEGAL_PARAM;
272  }
273 
274  // Pend on Semaphore.
275  params->status = Semaphore_pend(ECC_semaphore, params->timeout);
276 
277  // If execution returned due to a timeout
278  if (!params->status)
279  {
280  // Store status.
281  params->status = ECCROMCC26XX_STATUS_TIMEOUT;
282 
283  return ECCROMCC26XX_STATUS_TIMEOUT;
284  }
285 
286  // Allocate total memory for operation: workzone and 3 buffers
287  if (!(eccRom_workzone = (uint32_t *)params->malloc(params->curve.workzoneLen + ECC_BUF_TOTAL_LEN(params->curve.keyLen) * 3)))
288  {
289  // Post Semaphore.
290  Semaphore_post(ECC_semaphore);
291 
292  // Store status.
293  params->status = ECCROMCC26XX_STATUS_MALLOC_FAIL;
294 
295  return ECCROMCC26XX_STATUS_MALLOC_FAIL;
296  }
297 
298  // Split allocated memory into buffers
299  randStrBuf = (uint8_t *) eccRom_workzone + params->curve.workzoneLen;
300  pubKeyXBuf = randStrBuf + ECC_BUF_TOTAL_LEN(params->curve.keyLen);
301  pubKeyYBuf = pubKeyXBuf + ECC_BUF_TOTAL_LEN(params->curve.keyLen);
302 
303  // Initialize ECC curve and globals.
304  ECC_initGlobals(&params->curve);
305 
306  // Set length of keys in words in the first word of each buffer.
307  *((uint32_t *)&randStrBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
308  *((uint32_t *)&pubKeyXBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
309  *((uint32_t *)&pubKeyYBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
310 
311  // Copy in random string at key start offset.
312  memcpy(randStrBuf + ECC_KEY_OFFSET, privateKey, params->curve.keyLen);
313 
314  // Generate ECC private/public key pair.
315  // Note: the random string is an exact copy of the private key.
316  status = eccRom_genKeys((uint32_t *)randStrBuf,
317  (uint32_t *)randStrBuf,
318  (uint32_t *)pubKeyXBuf,
319  (uint32_t *)pubKeyYBuf);
320 
321  // Move ECC buffer values to client buffers.
322  memcpy(publicKeyX, pubKeyXBuf + ECC_KEY_OFFSET, params->curve.keyLen);
323  memcpy(publicKeyY, pubKeyYBuf + ECC_KEY_OFFSET, params->curve.keyLen);
324 
325  // zero out workzone and 3 buffers as a precautionary measure.
326  memset(eccRom_workzone, 0x00, params->curve.workzoneLen + params->curve.keyLen * 3);
327 
328  // Free allocated memory.
329  params->free((uint8_t *)eccRom_workzone);
330 
331  // Post Semaphore.
332  Semaphore_post(ECC_semaphore);
333 
334  // Map success code.
335  if (((uint8_t)status) == ECCROMCC26XX_STATUS_ECDH_KEYGEN_OK)
336  {
337  status = ECCROMCC26XX_STATUS_SUCCESS;
338  }
339 
340  // Store status.
341  params->status = status;
342 
343  return status;
344 }

361  if (privateKey == NULL || publicKeyX == NULL || publicKeyY == NULL ||

363  params->malloc == NULL || params->free == NULL)
364  {
365  // Store status.

384  }
385 
386  // Allocate total memory for operation: workzone and 5 key buffers.

388  {
389  // Post Semaphore.
390  Semaphore_post(ECC_semaphore);
391 
392  // Store status.
393  params->status = ECCROMCC26XX_STATUS_MALLOC_FAIL;
394 
395  return ECCROMCC26XX_STATUS_MALLOC_FAIL;
396  }

398  // Split allocated memory into buffers
399  privKeyBuf = (uint8_t *)eccRom_workzone + params->curve.workzoneLen;
400  pubKeyXBuf = privKeyBuf + ECC_BUF_TOTAL_LEN(params->curve.keyLen);
401  pubKeyYBuf = pubKeyXBuf + ECC_BUF_TOTAL_LEN(params->curve.keyLen);
402  DHKeyXBuf = pubKeyYBuf + ECC_BUF_TOTAL_LEN(params->curve.keyLen);
403  DHKeyYBuf = DHKeyXBuf + ECC_BUF_TOTAL_LEN(params->curve.keyLen);
404 
405  // Initialize ECC curve and globals.
406  ECC_initGlobals(&params->curve);
407 
408  // Set length of keys in words in the first word of each buffer.
409  *((uint32_t *)&privKeyBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
410  *((uint32_t *)&pubKeyXBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
411  *((uint32_t *)&pubKeyYBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
412  *((uint32_t *)&DHKeyXBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
413  *((uint32_t *)&DHKeyYBuf[ECC_KEY_LEN_OFFSET]) = (uint32_t)(ECC_UINT32_BLK_LEN(params->curve.keyLen));
414 
415  // Copy input keys into buffers.
416  memcpy(privKeyBuf + ECC_KEY_OFFSET, privateKey, params->curve.keyLen);
417  memcpy(pubKeyXBuf + ECC_KEY_OFFSET, publicKeyX, params->curve.keyLen);
418  memcpy(pubKeyYBuf + ECC_KEY_OFFSET, publicKeyY, params->curve.keyLen);
419 
420  status = ECC_VerifyPublicKeyWeierstrass(params, pubKeyXBuf, pubKeyYBuf,
421  params->curve.keyLen);
422 
423  if( status == ECCROMCC26XX_STATUS_SUCCESS)
424  {
425  // If the point is on curve generate shared key
426  // Generate ECC Diffie-Hellman shared secret.
427  status = eccRom_genSharedSecret((uint32_t *)privKeyBuf,
428  (uint32_t *)pubKeyXBuf,
429  (uint32_t *)pubKeyYBuf,
430  (uint32_t *)DHKeyXBuf,
431  (uint32_t *)DHKeyYBuf);
432 
433  // Move ECC buffer values to client buffers.
434  memcpy(dHKeyX, DHKeyXBuf + ECC_KEY_OFFSET, params->curve.keyLen);
435  memcpy(dHKeyY, DHKeyYBuf + ECC_KEY_OFFSET, params->curve.keyLen);
436  }
437 
438  // zero out workzone and 5 buffers as a precautionary measure.
439  memset(eccRom_workzone, 0x00, params->curve.workzoneLen + params->curve.keyLen * 5);
440 
441  // Free allocated memory.
442  params->free((uint8_t *)eccRom_workzone);
443 
444  // Post Semaphore.
445  Semaphore_post(ECC_semaphore);
446 
447  // Map success code.
448  if (((uint8_t)status) == ECCROMCC26XX_STATUS_ECDH_COMMON_KEY_OK)
449  {
450  status = ECCROMCC26XX_STATUS_SUCCESS;
451  }
452 
453  // Store status.
454  params->status = status;
455 
456  return status;
457 }

476 /*
477  * ======== ECC_ArrayAllZeros ========
478  */
479 bool ECC_ArrayAllZeros(const uint8_t *array, uint32_t arrayLength)
480 {
481  uint32_t i;
482  uint8_t arrayBits = 0;
483 
484  // We could speed things up by comparing word-wise rather than byte-wise.
485  // However, this extra overhead is inconsequential compared to running an
486  // actual PKA operation. Especially ECC operations.
487  for (i = 0; i < arrayLength; i++)
488  {
489  arrayBits |= array[i];
490  }
491 
492  if (arrayBits)
493  {
494  return FALSE;
495  }
496  else
497  {
498  return TRUE;
499  }
500 }

614

623  ((fn_threeOperandV_t)(mMULT_rom))(ECDSA_y2_hat, (uint32_t *)ECDSA_gy_mont, (uint32_t *)ECDSA_gy_mont);
624
