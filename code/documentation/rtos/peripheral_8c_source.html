277 /*********************************************************************
278  * @brief Set a GAP Role parameter.
279  *
280  * Public function defined in peripheral.h.
281  */
282 bStatus_t GAPRole_SetParameter(uint16_t param, uint8_t len, void *pValue)
283 {
284  bStatus_t ret = SUCCESS;
285  switch (param)
286  {
287  case GAPROLE_IRK:
288  if (len == KEYLEN)
289  {
290  VOID memcpy(gapRole_IRK, pValue, KEYLEN) ;
291  }
292  else
293  {
294  ret = bleInvalidRange;
295  }
296  break;
297 
298  case GAPROLE_SRK:
299  if (len == KEYLEN)
300  {
301  VOID memcpy(gapRole_SRK, pValue, KEYLEN) ;
302  }
303  else
304  {
305  ret = bleInvalidRange;
306  }
307  break;
308 
309  case GAPROLE_SIGNCOUNTER:
310  if (len == sizeof (uint32_t))
311  {
312  gapRole_signCounter = *((uint32_t*)pValue);
313  }
314  else
315  {
316  ret = bleInvalidRange;
317  }
318  break;
319 
320  case GAPROLE_ADVERT_ENABLED:
321  if (len == sizeof(uint8_t))
322  {
323  // Non-connectable advertising must be disabled.
324  if (gapRole_AdvNonConnEnabled != TRUE)
325  {
326  uint8_t oldAdvEnabled = gapRole_AdvEnabled;
327  gapRole_AdvEnabled = *((uint8_t*)pValue);
328 
329  if ((oldAdvEnabled) && (gapRole_AdvEnabled == FALSE))
330  {
331  // Turn off advertising.
332  if ((gapRole_state == GAPROLE_ADVERTISING)
333  || (gapRole_state == GAPROLE_WAITING_AFTER_TIMEOUT))
334  {
335  VOID GAP_EndDiscoverable(selfEntity);
336  }
337  }
338  else if ((oldAdvEnabled == FALSE) && (gapRole_AdvEnabled))
339  {
340  // Turn on advertising.
341  if ((gapRole_state == GAPROLE_STARTED)
342  || (gapRole_state == GAPROLE_WAITING)
343  || (gapRole_state == GAPROLE_WAITING_AFTER_TIMEOUT))
344  {
345  gapRole_setEvent(START_ADVERTISING_EVT);
346  }
347  }
348  }
349  else
350  {
351  ret = bleIncorrectMode;
352  }
353  }
354  else
355  {
356  ret = bleInvalidRange;
357  }
358  break;
359 
360  case GAPROLE_ADV_NONCONN_ENABLED:
361  if (len == sizeof(uint8_t))
362  {
363  // Connectable advertising must be disabled.
364  if (gapRole_AdvEnabled != TRUE)
365  {
366  uint8_t oldAdvEnabled = gapRole_AdvNonConnEnabled;
367  gapRole_AdvNonConnEnabled = *((uint8_t*)pValue);
368 
369  if ((oldAdvEnabled) && (gapRole_AdvNonConnEnabled == FALSE))
370  {
371  if ((gapRole_state == GAPROLE_ADVERTISING_NONCONN)
372  || (gapRole_state == GAPROLE_CONNECTED_ADV)
373  || (gapRole_state == GAPROLE_WAITING_AFTER_TIMEOUT))
374  {
375  VOID GAP_EndDiscoverable(selfEntity);
376  }
377  }
378  else if ((oldAdvEnabled == FALSE) && (gapRole_AdvNonConnEnabled))
379  {
380  // Turn on advertising.
381  if ((gapRole_state == GAPROLE_STARTED)
382  || (gapRole_state == GAPROLE_WAITING)
383  || (gapRole_state == GAPROLE_CONNECTED)
384  || (gapRole_state == GAPROLE_WAITING_AFTER_TIMEOUT))
385  {
386  gapRole_setEvent(START_ADVERTISING_EVT);
387  }
388  }
389  }
390  else
391  {
392  ret = bleIncorrectMode;
393  }
394  }
395  else
396  {
397  ret = bleInvalidRange;
398  }
399  break;
400 
401  case GAPROLE_ADVERT_OFF_TIME:
402  if (len == sizeof (uint16_t))
403  {
404  gapRole_AdvertOffTime = *((uint16_t*)pValue);
405  }
406  else
407  {
408  ret = bleInvalidRange;
409  }
410  break;
411 
412  case GAPROLE_ADVERT_DATA:
413  if (len <= B_MAX_ADV_LEN)
414  {
415  VOID memset(gapRole_AdvertData, 0, B_MAX_ADV_LEN);
416  VOID memcpy(gapRole_AdvertData, pValue, len);
417  gapRole_AdvertDataLen = len;
418 
419  // Update the advertising data
420  ret = GAP_UpdateAdvertisingData(selfEntity,
421  TRUE, gapRole_AdvertDataLen, gapRole_AdvertData);
422  }
423  else
424  {
425  ret = bleInvalidRange;
426  }
427  break;
428 
429  case GAPROLE_SCAN_RSP_DATA:
430  if (len <= B_MAX_ADV_LEN)
431  {
432  VOID memset(gapRole_ScanRspData, 0, B_MAX_ADV_LEN);
433  VOID memcpy(gapRole_ScanRspData, pValue, len);
434  gapRole_ScanRspDataLen = len;
435 
436  // Update the Response Data
437  ret = GAP_UpdateAdvertisingData(selfEntity,
438  FALSE, gapRole_ScanRspDataLen, gapRole_ScanRspData);
439  }
440  else
441  {
442  ret = bleInvalidRange;
443  }
444  break;
445 
446  case GAPROLE_ADV_EVENT_TYPE:
447  if ((len == sizeof (uint8_t)) && (*((uint8_t*)pValue) <= GAP_ADTYPE_ADV_LDC_DIRECT_IND))
448  {
449  gapRole_AdvEventType = *((uint8_t*)pValue);
450  }
451  else
452  {
453  ret = bleInvalidRange;
454  }
455  break;
456 
457  case GAPROLE_ADV_DIRECT_TYPE:
458  if ((len == sizeof (uint8_t)) && (*((uint8_t*)pValue) <= ADDRMODE_PRIVATE_RESOLVE))
459  {
460  gapRole_AdvDirectType = *((uint8_t*)pValue);
461  }
462  else
463  {
464  ret = bleInvalidRange;
465  }
466  break;
467 
468  case GAPROLE_ADV_DIRECT_ADDR:
469  if (len == B_ADDR_LEN)
470  {
471  VOID memcpy(gapRole_AdvDirectAddr, pValue, B_ADDR_LEN) ;
472  }
473  else
474  {
475  ret = bleInvalidRange;
476  }
477  break;
478 
479  case GAPROLE_ADV_CHANNEL_MAP:
480  if ((len == sizeof (uint8_t)) && (*((uint8_t*)pValue) <= 0x07))
481  {
482  gapRole_AdvChanMap = *((uint8_t*)pValue);
483  }
484  else
485  {
486  ret = bleInvalidRange;
487  }
488  break;
489 
490  case GAPROLE_ADV_FILTER_POLICY:
491  if ((len == sizeof (uint8_t)) && (*((uint8_t*)pValue) <= GAP_FILTER_POLICY_WHITE))
492  {
493  gapRole_AdvFilterPolicy = *((uint8_t*)pValue);
494  }
495  else
496  {
497  ret = bleInvalidRange;
498  }
499  break;
500 
501  case GAPROLE_PARAM_UPDATE_ENABLE:
502  if ((len == sizeof (uint8_t)) &&
503  (*((uint8_t*)pValue) < GAPROLE_LINK_PARAM_UPDATE_NUM_OPTIONS))
504  {
505  gapRole_updateConnParams.paramUpdateEnable = *((uint8_t*)pValue);
506  }
507  else
508  {
509  ret = bleInvalidRange;
510  }
511  break;
512 
513  case GAPROLE_MIN_CONN_INTERVAL:
514  {
515  uint16_t newInterval = *((uint16_t*)pValue);
516  if ( len == sizeof (uint16_t) &&
517  (newInterval >= MIN_CONN_INTERVAL) &&
518  (newInterval <= MAX_CONN_INTERVAL))
519  {
520  gapRole_updateConnParams.minConnInterval = newInterval;
521  }
522  else
523  {
524  ret = bleInvalidRange;
525  }
526  }
527  break;
528 
529  case GAPROLE_MAX_CONN_INTERVAL:
530  {
531  uint16_t newInterval = *((uint16_t*)pValue);
532  if ( len == sizeof (uint16_t) &&
533  (newInterval >= MIN_CONN_INTERVAL) &&
534  (newInterval <= MAX_CONN_INTERVAL))
535  {
536  gapRole_updateConnParams.maxConnInterval = newInterval;
537  }
538  else
539  {
540  ret = bleInvalidRange;
541  }
542  }
543  break;
544 
545  case GAPROLE_SLAVE_LATENCY:
546  {
547  uint16_t latency = *((uint16_t*)pValue);
548  if (len == sizeof (uint16_t) && (latency < MAX_SLAVE_LATENCY))
549  {
550  gapRole_updateConnParams.slaveLatency = latency;
551  }
552  else
553  {
554  ret = bleInvalidRange;
555  }
556  }
557  break;
558 
559  case GAPROLE_TIMEOUT_MULTIPLIER:
560  {
561  uint16_t newTimeout = *((uint16_t*)pValue);
562  if (len == sizeof (uint16_t)
563  && (newTimeout >= MIN_TIMEOUT_MULTIPLIER) && (newTimeout <= MAX_TIMEOUT_MULTIPLIER))
564  {
565  gapRole_updateConnParams.timeoutMultiplier = newTimeout;
566  }
567  else
568  {
569  ret = bleInvalidRange;
570  }
571  }
572  break;
573 
574  case GAPROLE_PARAM_UPDATE_REQ:
575  {
576  uint8_t req = *((uint8_t*)pValue);
577  if (len == sizeof (uint8_t) && (req == TRUE))
578  {
579  // Make sure we don't send an L2CAP Connection Parameter Update Request
580  // command within TGAP(conn_param_timeout) of an L2CAP Connection Parameter
581  // Update Response being received.
582  if (Util_isActive(&updateTimeoutClock) == FALSE)
583  {
584  // Start connection update procedure
585  ret = gapRole_startConnUpdate(GAPROLE_NO_ACTION, &gapRole_updateConnParams);
586  if (ret == SUCCESS)
587  {
588  // Connection update requested by app, cancel such pending procedure (if active)
589  Util_stopClock(&startUpdateClock);
590  }
591  }
592  else
593  {
594  ret = blePending;
595  }
596  }
597  else
598  {
599  ret = bleInvalidRange;
600  }
601  }
602  break;
603 
604  default:
605  // The param value isn't part of this profile, try the GAP.
606  if ((param < TGAP_PARAMID_MAX) && (len == sizeof (uint16_t)))
607  {
608  ret = GAP_SetParamValue(param, *((uint16_t*)pValue));
609  }
610  else
611  {
612  ret = INVALIDPARAMETER;
613  }
614  break;
615  }
616 
617  return (ret);
618 }

1100  VOID osal_snv_write(BLE_NVID_CSRK, KEYLEN, gapRole_SRK);
1101 
1102  // Save off the information
1103  VOID memcpy(gapRole_bdAddr, pPkt->devAddr, B_ADDR_LEN);
1104

1106

1109  TRUE, gapRole_AdvertDataLen, gapRole_AdvertData);
1110  }
1111

1115  }
1116 
1117  notify = TRUE;
1118  }
1119  break;

1121  case GAP_ADV_DATA_UPDATE_DONE_EVENT:
1122  {
1123  gapAdvDataUpdateEvent_t *pPkt = (gapAdvDataUpdateEvent_t *)pMsg;
1124

1126  {
1127  if (pPkt->adType)
1128  {
1129  // Setup the Response Data
1130  pPkt->hdr.status = GAP_UpdateAdvertisingData(selfEntity, FALSE,
1131  gapRole_ScanRspDataLen,
1132  gapRole_ScanRspData);
1133  }
1134  else if ((gapRole_state != GAPROLE_ADVERTISING) &&
1135  (gapRole_state != GAPROLE_CONNECTED_ADV) &&
1136  (gapRole_state != GAPROLE_CONNECTED ||
1137  gapRole_AdvNonConnEnabled == TRUE) &&
1138  (gapRole_state != GAPROLE_ADVERTISING_NONCONN) &&
1139  (Util_isActive(&startAdvClock) == FALSE))
1140  {
1141  // Start advertising
1142  gapRole_setEvent(START_ADVERTISING_EVT);
1143  }
1144  }
1145

1147  {
1148  // Set into Error state
1149  gapRole_state = GAPROLE_ERROR;
1150  notify = TRUE;
1151  }
1152  }
1153  break;
1154 
1155  case GAP_MAKE_DISCOVERABLE_DONE_EVENT:
1156  case GAP_END_DISCOVERABLE_DONE_EVENT:

1159

1161  {

1163  {

1165  {
1166  gapRole_state = GAPROLE_CONNECTED_ADV;
1167  }
1168  else if (gapRole_AdvEnabled)
1169  {
1170  gapRole_state = GAPROLE_ADVERTISING;
1171  }
1172  else
1173  {
1174  gapRole_state = GAPROLE_ADVERTISING_NONCONN;
1175  }
1176  }
1177  else // GAP_END_DISCOVERABLE_DONE_EVENT
1178  {
1179  if (gapRole_AdvertOffTime != 0)
1180  {
1181  if ((gapRole_AdvEnabled) || (gapRole_AdvNonConnEnabled))
1182  {
1183  Util_restartClock(&startAdvClock, gapRole_AdvertOffTime);
1184  }
1185  }
1186  else

1189  // automatically become discoverable again after a period of time.
1190  // Set enabler to FALSE; device will become discoverable again when
1191  // this value gets set to TRUE
1192  if (gapRole_AdvEnabled == TRUE)
1193  {
1194  gapRole_AdvEnabled = FALSE;
1195  }
1196  else
1197  {
1198  gapRole_AdvNonConnEnabled = FALSE;
1199  }
1200  }
1201

1203  if (gapRole_state == GAPROLE_CONNECTED_ADV)
1204  {
1205  // In the Advertising Off period
1206  gapRole_state = GAPROLE_CONNECTED;
1207  }
1208  else
1209  {
1210  // In the Advertising Off period
1211  gapRole_state = GAPROLE_WAITING;
1212  }
1213  }
1214  }
1215  else
1216  {
1217  gapRole_state = GAPROLE_ERROR;
1218  }
1219 
1220  notify = TRUE;
1221  }

1227

1229  {
1230  VOID memcpy(gapRole_ConnectedDevAddr, pPkt->devAddr, B_ADDR_LEN);
1231  gapRole_ConnectionHandle = pPkt->connectionHandle;
1232  gapRole_state = GAPROLE_CONNECTED;
1233 
1234  // Store connection information

1236  gapRole_ConnSlaveLatency = pPkt->connLatency;
1237  gapRole_ConnTimeout = pPkt->connTimeout;

1391  // connection parameters given to them by the remote device." The
1392  // interpretation here is to check if the connection parameters
1393  // given by the remote device are compatible with the desired values

1400  (pReq->req.intervalMin < gapRole_updateConnParams.minConnInterval) ?

1408  pReq->req.intervalMax;
1409

1411  rsp.connLatency =
1412  (pReq->req.connLatency < gapRole_updateConnParams.slaveLatency) ?
1413  gapRole_updateConnParams.slaveLatency :
1414  pReq->req.connLatency;
1415
