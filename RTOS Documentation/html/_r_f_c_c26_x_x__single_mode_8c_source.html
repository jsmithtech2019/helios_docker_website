306 /* Push entry onto simple queue */
307 static bool Q_push(RF_Cmd* volatile* ppHead, RF_Cmd* pCmd)
308 {
309  RF_Cmd* pEnd = (RF_Cmd*)*ppHead;
310  uint32_t key;
311 
312  pCmd->pNext = NULL;
313  if (!pEnd)
314  {
315  // First entry into queue
316  key = Hwi_disable();
317  *ppHead = pCmd;
318  }
319  else
320  {
321  key = Hwi_disable();
322  while (pEnd->pNext)
323  {
324  pEnd = (RF_Cmd*)pEnd->pNext;
325  }
326 
327  // Sanity check - to avoid pEnd->pNext pointing to self
328  if (pEnd == pCmd)
329  {
330  Hwi_restore(key);
331  return false;
332  }
333 
334  pEnd->pNext = pCmd;
335  }
336  Hwi_restore(key);
337 
338  return true;
339 }

341 /* Search last entry in simple queue for particular client */
342 static RF_Cmd* Q_end(RF_Handle h, RF_Cmd* volatile* ppHead)
343 {
344  RF_Cmd* pEnd = NULL;
345  RF_Cmd* pClientEnd = NULL;
346 
347  uint32_t key = Hwi_disable();
348  if ((pEnd = (RF_Cmd*)*ppHead))
349  {
350  pClientEnd = pEnd;
351  while (pEnd->pNext)
352  {
353  pEnd = (RF_Cmd*)pEnd->pNext;
354  pClientEnd = pEnd;
355  }
356  }
357  Hwi_restore(key);
358 
359  return pClientEnd;
360 }

414 /*
415  * Function: dispatchOpWhen
416  * Check whether to dispatch operation immediately or power down radio and wait.
417  * If the former it returns 0, if the latter it returns the #us until when RF
418  * need to execute the command
419  *
420  * Calling context: internal (Hwi, Swi)
421  *
422  * Input: pOp - Pointer to radio operation command structure
423  *
424  * Return: Time (in us) till radio command execution
425  */
426 static uint32_t dispatchOpWhen(RF_Op* pOp)
427 {
428  // Limitations:
429  // - Only recognizes TRIG_ABSTIME triggers, everything else gets dispatched immediately
430  if (pOp->startTrigger.triggerType == TRIG_ABSTIME)
431  {
432  // Difference of startTime from current time in us
433  uint32_t deltaTimeUs = (pOp->startTime - RF_getCurrentTime()) / RF_NUM_RAT_TICKS_IN_1_US;
434 
435  // Check if delta time is greater than (powerup duration + power down duration) for a
436  // power cycle, and is less than 3/4 of a RAT cycle (~17 minutes)
437  if ((deltaTimeUs > (int32_t)(pCurrClient->clientConfig.nPowerUpDuration + RF_DEFAULT_POWER_DOWN_TIME)) &&
438  (deltaTimeUs <= RF_DISPATCH_MAX_TIME_US))
439  {
440  // Delta time is within range
441  return deltaTimeUs;
442  }
443  }
444  return 0;
445 }

491 // SWI used for user callback to any RAT event (RF_HW HWI)
492 static void swiFxnHw(xdc_UArg a, xdc_UArg b)
493 {
494  RF_CmdDirImm* pDirImmCmd = &cmdDirImm;
495 
496  // RAT channel callback
497  // Optimize code to issue callback in the order in which the RAT ch int was received
498  uint32_t key = Hwi_disable();
499  if (pDirImmCmd->hwIntEvent & RF_HW_INT_RAT_CH_MASK)
500  {
501  uint8_t ratCbIdx = pDirImmCmd->hwIntEvent >> RFC_DBELL_RFHWIFG_RATCH5_BITN;
502  // Process lower channel first and allow multiple interrupt flags to be processed sequentially
503  if (ratCbIdx & (1 << RF_RAT_CH_5_INDEX))
504  {
505  ratCbIdx = RF_RAT_CH_5_INDEX;
506  }
507  else if (ratCbIdx & (1 << RF_RAT_CH_6_INDEX))
508  {
509  ratCbIdx = RF_RAT_CH_6_INDEX;
510  }
511  else if (ratCbIdx & (1 << RF_RAT_CH_7_INDEX))
512  {
513  ratCbIdx = RF_RAT_CH_7_INDEX;
514  }
515 
516  ratChanFree(ratCbIdx + RF_RAT_CH_OFFSET);
517  pDirImmCmd->hwIntEvent &= ~(1 << (ratCbIdx + RFC_DBELL_RFHWIFG_RATCH5_BITN));
518  Hwi_restore(key);
519  if (pDirImmCmd->pRatCb[ratCbIdx])
520  {
521  pDirImmCmd->pRatCb[ratCbIdx](pDirImmCmd->pClient, ratCbIdx + RF_RAT_CH_OFFSET, RF_EventRatCh);
522  }
523  }
524  else
525  {
526  Hwi_restore(key);
527  }
528 
529  // Process if multiple interrupt flag bits are set
530  if (pDirImmCmd->hwIntEvent)
531  {
532  Swi_or(Swi_handle(&swiHw), 0);
533  }
534 }

536 // The CPE0 ISR when radio is active. Assume that all IRQs relevant to command
537 // dispatcher are mapped here. Furthermore, assume that there is no need for
538 // critical sections here (i.e. that this ISR has higher priority than
539 // any HWI calling a RF API function or that HWIs can't call the RF API)
540 static void hwiFxnCpe0Active(xdc_UArg a)
541 {
542  RF_Cmd* pCmd;
543 
544  // Get and clear doorbell IRQs
545  uint32_t rfcpeifg = RFCCpeIntGetAndClear();
546 
547  // Retire command that has run
548  if (rfcpeifg & RFC_DBELL_RFCPEIFG_LAST_COMMAND_DONE_M)
549  {
550  // LAST_COMMAND_DONE received.
551  if ((pCmd = cmdQ.pCurrCmd))
552  {
553  // No longer current command
554  cmdQ.pCurrCmd = NULL;
555 
556  // Save CPEIFG for callback
557  pCmd->rfifg |= ((rfcpeifg & pCmd->bmEvent) | RF_EventLastCmdDone);
558 
559  // Move current command to done queue
560  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFCPEIEN) &= ~(pCmd->bmEvent & RF_INTERNAL_CPE_IFG_MASK_32_BIT);
561  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFHWIEN) &= ~(pCmd->bmEvent >> RF_SHIFT_32_BITS);
562  Q_push(&cmdQ.pDone, pCmd);
563 
564  // Post SWI to handle registered callbacks and potentially start inactivity timer
565  Swi_or(Swi_handle(&swiFsm), Fsm_EventLastCommandDone);
566  }
567  }
568 
569  // Handle additional interrupts here?
570  pCmd = cmdQ.pCurrCmd;
571  if ((!(rfcpeifg & Fsm_EventLastCommandDone)) && (rfcpeifg & (pCmd->bmEvent)))
572  {
573  cmdQ.pCurrCmdCb = pCmd;
574 
575  // Save CPEIFG for callback
576  pCmd->rfifg |= (rfcpeifg & pCmd->bmEvent);
577 
578  // Post SWI to handle registered callbacks
579  Swi_or(Swi_handle(&swiFsm), Fsm_EventCpeInt);
580  }
581 
582  // Dispatch pending command
583  if ((!cmdQ.pCurrCmd) && (pCmd = Q_peek(&cmdQ.pPend)))
584  {

586  {

588  // changed (either as part of power-up or PHY change) the FSM will invoke
589  // this ISR and dispatch command (Unsupported feature)
590  Swi_or(Swi_handle(&swiFsm), Fsm_EventChangePhy);
591  }
592  else if ((dispatchInUs = dispatchOpWhen(pCmd->pOp)))
593  {
594  // Command sufficiently far into future that it shouldn't be dispatched yet
595  // Decrement active client count and potentially power down radio
596  decActiveClientCnt(dispatchInUs);
597  }
598  else
599  {
600  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFCPEIEN) |= (pCmd->bmEvent & RF_INTERNAL_CPE_IFG_MASK_32_BIT);
601  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFHWIEN) |= (pCmd->bmEvent >> RF_SHIFT_32_BITS);
602 
603  // Dispatch immediately, set command to current and remove from pending queue

605  HWREG(RFC_DBELL_BASE+RFC_DBELL_O_CMDR) = (uint32_t)pCmd->pOp;
606  while(!HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG));
607 
608  Q_pop(&cmdQ.pPend);
609  cmdQ.pCurrCmd = pCmd;
610  }
611  }

725 /*
726  * Function: radioOpDoneCb
727  *
728  * Handle callback to client for RF_EventLastCmdDone and issue radio free callback if required
729  *
730  */
731 static void radioOpDoneCb(RF_Cmd* pCmd)
732 {
733  /* Radio command done */
734  if (pCmd)
735  {
736  // Update implicit radio state
737  updateImplicitState(pCmd);
738 
739  uint32_t tmp = pCmd->rfifg;
740  pCmd->rfifg = 0;
741  // Call callback, free container and dequeue
742  if (pCmd->pCb)
743  {
744  RF_EventMask cb_e = (RF_EventMask)(tmp | RF_EventLastCmdDone);
745  switch (pCmd->flags & RF_CMD_CANCEL_MASK)
746  {
747  case RF_CMD_ABORTED_FLAG:
748  cb_e = RF_EventCmdAborted;
749  break;
750  case RF_CMD_STOPPED_FLAG:
751  cb_e = RF_EventCmdStopped;
752  break;
753  case RF_CMD_CANCELLED_FLAG:
754  cb_e = RF_EventCmdCancelled;
755  break;
756  default:
757  break;
758  }
759  pCmd->pCb(pCmd->pClient, pCmd->ch, cb_e);
760  }
761 
762  uint32_t key = Hwi_disable();
763  cmdQ.nSeqDone = (cmdQ.nSeqDone+1)&N_CMD_MODMASK;
764  cmdQ.pDone->flags = 0;
765  Q_pop(&cmdQ.pDone);
766  Hwi_restore(key);
767 
768  // Check if there are any more pending commands for this client
769  if ((cmdQ.nSeqDone == pCmd->pClient->state.chLastPosted) && (!bRatActive))
770  {
771  // This was last posted command for client
772  if (pCmd->pClient->state.bYielded || (pCmd->pClient->clientConfig.nInactivityTimeout == 0))
773  {
774  // We can powerdown immediately
775  pCmd->pClient->state.bYielded = false;
776  decActiveClientCnt(0);
777  }
778  else
779  {
780  if (pCurrClient->clientConfig.nInactivityTimeout != BIOS_WAIT_FOREVER)
781  {
782  // Start inactivity timer
783  uint32_t timeout = MAX(pCmd->pClient->clientConfig.nInactivityTimeout/Clock_tickPeriod, 1);
784  Clock_setTimeout(Clock_handle(&pCmd->pClient->state.clkInactivity), timeout);
785  Clock_start(Clock_handle(&pCmd->pClient->state.clkInactivity));
786  }
787  }
788  }
789  }

792 static void fsmPowerUpState(RF_Object *pObj, Fsm_Event e)
793 {
794  // Note: pObj is NULL in this state
795  if (e & Fsm_EventLastCommandDone)
796  {
797  /* issue radio operation done callback */
798  RF_Cmd *pCmd = Q_peek(&cmdQ.pDone);
799  radioOpDoneCb(pCmd);
800 
801  /* We've handled this event now */
802  e &= ~Fsm_EventLastCommandDone;
803  if ((pCmd = Q_peek(&cmdQ.pDone)))
804  {
805  /* Trigger self if there are more commands in callback queue */
806  Swi_or(Swi_handle(&swiFsm), (e | Fsm_EventLastCommandDone));
807  }
808  else if ((e & Fsm_EventWakeup) || (Q_peek(&cmdQ.pPend)))
809  {
810  /* If there is pending cmd, just wake up */
811  /* Device will put to sleep if the pending cmd is future cmd */
812  Clock_stop(Clock_handle(&clkPowerUp));
813 
814  /* trigger power up */
815  Swi_or(Swi_handle(&swiFsm), Fsm_EventWakeup);
816  }
817  }
818  else if(e & Fsm_EventWakeup)
819  {
820  // Set current client from first command in command queue
821  RF_Cmd* pNextCmd;
822  if ((pNextCmd = Q_peek(&cmdQ.pPend)))
823  {
824  pCurrClient = pNextCmd->pClient;
825  }
826 
827  // Keep track of mode
828  HWREG(PRCM_BASE + PRCM_O_RFCMODESEL) = pCurrClient->clientConfig.pRfMode->rfMode;
829  if (HWREG(PRCM_BASE + PRCM_O_RFCMODESEL) != pCurrClient->clientConfig.pRfMode->rfMode)
830  {
831  // Unsupported mode, do error callback
832  if (pCurrClient->clientConfig.pErrCb)
833  {
834  RF_Callback pErrCb = (RF_Callback)pCurrClient->clientConfig.pErrCb;
835  pErrCb(pCurrClient, RF_ERROR_INVALID_RFMODE, RF_EventError);
836  }
837 
838  }
839 
840  // Register power dependency and constraints - i.e. power up RF.
841  Power_setConstraint(PowerCC26XX_SB_DISALLOW);
842  Power_setDependency(PowerCC26XX_DOMAIN_RFCORE);
843 
844  // Start power-up sequence
845  bPowerUpActive = true;
846 
847  if((pCurrClient->clientConfig.bPowerUpXOSC == false) || bStart == 0)
848  {
849  Power_setDependency(PowerCC26XX_XOSC_HF);
850  }
851 
852  // Turn on additional clocks on boot
853  if ((pCurrClient->clientConfig.pRfMode->mcePatchFxn != 0) || (pCurrClient->clientConfig.pRfMode->rfePatchFxn != 0))
854  {
855  HWREG(RFC_DBELL_BASE + RFC_DBELL_O_RFACKIFG) = 0;
856  HWREG(RFC_DBELL_BASE+RFC_DBELL_O_CMDR) = CMDR_DIR_CMD_2BYTE(RF_CMD0, RFC_PWR_PWMCLKEN_MDMRAM | RFC_PWR_PWMCLKEN_RFERAM);
857  }
858 
859  // Turn on clock to RF core
860  RFCClockEnable();
861 
862  // Map all radio interrupts to CPE0 and enable some sources
863  HWREG(RFC_DBELL_BASE+RFC_DBELL_O_RFCPEISL) = 0;
864 
865  HWREG(RFC_DBELL_BASE+RFC_DBELL_O_RFCPEIEN) =
866  RFC_DBELL_RFCPEIEN_LAST_COMMAND_DONE_M |
867  RFC_DBELL_RFCPEIEN_BOOT_DONE_M;
868 
869  // Next FSM state
870  FsmNextState(&fsmSetupState);
871 
872  // Enable interrupts and let BOOT_DONE interrupt kick things off
873  Hwi_enableInterrupt(INT_RFC_CPE_0);
874  }

1002 static void fsmActiveState(RF_Object *pObj, Fsm_Event e)
1003 {
1004  RF_Cmd* pCmd;
1005  RF_CmdDirImm* pCmdDirImm;
1006 
1007  if (e & Fsm_EventCpeInt)
1008  {
1009  uint32_t key = Hwi_disable();
1010  pCmd = Q_peek(&cmdQ.pCurrCmdCb);
1011  uint32_t tmp = pCmd->rfifg;
1012 
1013  if (!(tmp & RF_EventLastCmdDone) && pCmd)
1014  {
1015  pCmd->rfifg &= (~tmp);
1016  Hwi_restore(key);
1017  if (pCmd->pCb)
1018  {
1019  pCmd->pCb(pCmd->pClient, pCmd->ch, (RF_EventMask)tmp);
1020  }
1021  }
1022  else
1023  {
1024  Hwi_restore(key);
1025  }
1026 
1027  // We've handled this event now
1028  e &= ~Fsm_EventCpeInt;
1029  }
1030  else if (e & Fsm_EventLastCommandDone)
1031  {
1032  /* issue radio operation done callback */
1033  pCmd = Q_peek(&cmdQ.pDone);
1034  radioOpDoneCb(pCmd);
1035 
1036  // We've handled this event now
1037  e &= ~Fsm_EventLastCommandDone;
1038  if ((pCmd = Q_peek(&cmdQ.pDone)))
1039  {
1040  // Trigger self if there are more commands in callback queue
1041  e |= Fsm_EventLastCommandDone;
1042  }
1043  }
1044  else if (e & Fsm_EventPowerStep)
1045  {
1046  // Coming from powerup states
1047 
1048  // For power saving, set analogCfgMode to 0x2D
1049  if (pObj->clientConfig.pOpSetup->commandId.commandNo == CMD_RADIO_SETUP)
1050  {
1051  pObj->clientConfig.pOpSetup->common.config.analogCfgMode = RF_SETUP_ANALOGCFG_NOUPDATE;
1052  }
1053  else
1054  {
1055  pObj->clientConfig.pOpSetup->prop.config.analogCfgMode = RF_SETUP_ANALOGCFG_NOUPDATE;
1056  }
1057 
1058  // update power up duration if coming from the clkPowerUpFxn (nRtc not zero)
1059  uint32_t nRtc2;
1060  if (nRtc1 && (OSCClockSourceGet(OSC_SRC_CLK_LF) != OSC_RCOSC_HF))
1061  {
1062  nRtc2 = AONRTCCurrent64BitValueGet();
1063  if (nRtc2 > nRtc1)
1064  {
1065  pObj->clientConfig.nPowerUpDuration = (nRtc2 - nRtc1) >> RF_RTC_CONV_TO_US_SHIFT;
1066  }
1067  else
1068  {
1069  pObj->clientConfig.nPowerUpDuration = (0xFFFFFFFF + nRtc2 - nRtc1) >> RF_RTC_CONV_TO_US_SHIFT;
1070  }
1071 
1072  // add margin to the power up duration to account for the TIRTOS/Drivers overhead
1073  pObj->clientConfig.nPowerUpDuration += pObj->clientConfig.nPowerUpDurationMargin;
1074 
1075  // power up duration should be within certain upper and lower bounds
1076  if ((pObj->clientConfig.nPowerUpDuration > RF_DEFAULT_POWER_UP_TIME) ||
1077  (pObj->clientConfig.nPowerUpDuration < RF_DEFAULT_MIN_POWER_UP_TIME))
1078  {
1079  pObj->clientConfig.nPowerUpDuration = RF_DEFAULT_POWER_UP_TIME;
1080  }
1081  }
1082 
1083  // Check the status of the CMD_FS, if it was sent (chained) to the setup command sent
1084  // during Fsm_EventChangePhy1 and return error callback to the client
1085  // client can either resend the CMD_FS or ignore the error as per Errata on PG2.1
1086  // As this is a recoverable event do not call default err callback : while(1)
1087  if ((pObj->clientConfig.pErrCb) && ((RF_Callback)pObj->clientConfig.pErrCb != defaultErrCb))
1088  {
1089  if ((opRatSync.start.condition.rule == COND_ALWAYS) &&
1090  (pObj->state.mode_state.cmdFs.status == ERROR_SYNTH_PROG))
1091  {
1092  RF_Callback pErrCb = (RF_Callback)pObj->clientConfig.pErrCb;
1093  pErrCb(pObj, RF_ERROR_CMDFS_SYNTH_PROG, RF_EventError);
1094  }
1095  }
1096 
1097  // Change HWI handler
1098  Hwi_setFunc(Hwi_handle(&hwiCpe0), hwiFxnCpe0Active, (UArg)NULL);
1099 
1100  // Get the available RAT channel map
1101  pCmdDirImm = &cmdDirImm;
1102  if (bInitAvailRatCh)
1103  {
1104  rfc_CMD_GET_FW_INFO_t getFwInfoCmd;
1105  getFwInfoCmd.commandNo = CMD_GET_FW_INFO;
1106  RFCDoorbellSendTo((uint32_t)&getFwInfoCmd);
1107  pCmdDirImm->availRatCh = getFwInfoCmd.availRatCh;
1108  bInitAvailRatCh = false;
1109  }
1110 
1111  // Mark radio and client as being active
1112  bRadioActive = true;
1113  nActiveClients++;
1114  bPowerUpActive = false;
1115 
1116  // Check if there is pending command
1117  if (Q_peek(&cmdQ.pPend))
1118  {
1119  // Make sure pending command gets issued by issuing CPE0 IRQ
1120  Hwi_post(INT_RFC_CPE_0);
1121  }
1122  else
1123  {
1124  // No pending command
1125  e |= Fsm_EventLastCommandDone;
1126  }
1127 
1128  // Power up callback
1129  if (pObj->clientConfig.pPowerCb)
1130  {
1131  RF_Callback pPowerCb = (RF_Callback)pObj->clientConfig.pPowerCb;
1132  pPowerCb(pObj, 0, RF_EventPowerUp);
1133  }
1134 
1135  // We've handled this event now
1136  e &= ~Fsm_EventPowerStep;
1137  }
1138  else if (bYieldPwrDnCancel && (e & Fsm_EventPowerDown))
1139  {
1140  bYieldPwrDnCancel = false;
1141  e &= ~Fsm_EventPowerDown;
1142  }
1143  else if (e & Fsm_EventPowerDown)
1144  {
1145  // Powerdown RF core
1146  // The proper thing to do here is to avoid switching off the digital part of
1147  // radio without switching off the analog parts first
1148 
1149  // Radio not active
1150  bRadioActive = false;
1151  bPowerDownActive = true;
1152 
1153  // set VCOLDO reference
1154  RFCAdi3VcoLdoVoltageMode(false);
1155 
1156  // Next state: fsmPowerDown2State
1157  FsmNextState(&fsmPowerDownState);
1158 
1159  // Remap HWI to startup function
1160  Hwi_setFunc(Hwi_handle(&hwiCpe0), hwiFxnCpe0PowerFsm, (UArg)NULL);
1161 
1162  if ((bStart == 0) || (bInitAvailRatCh))
1163  {
1164  bStart = 1;
1165 
1166  // Always do RAT_SYNC_STOP command the first time power down
1167  opRatSync.stop.commandNo = CMD_SYNC_STOP_RAT;
1168  opRatSync.stop.condition.rule = COND_NEVER;
1169  opRatSync.stop.status = IDLE;
1170  opRatSync.stop.startTrigger.triggerType = TRIG_NOW;
1171  opRatSync.stop.pNextOp = NULL;
1172  HWREG(RFC_DBELL_BASE+RFC_DBELL_O_CMDR) = (uint32_t)&opRatSync.stop;
1173  }
1174  else if (dispatchInUs)
1175  {

1201  }

1204  nRtc1 = 0;

1220  {

1222  }

1538  Assert_isTrue((pOp != NULL), NULL);
1539 
1540 
1541  uint32_t key = Hwi_disable();
1542

1544  if (((cmdQ.nSeqPost + N_CMD_MODMASK + 1 - cmdQ.nSeqDone) & N_CMD_MODMASK) < N_CMD_POOL)
1545  {
1546  // Allocate RF_Op container
1547  cmdQ.nSeqPost = (cmdQ.nSeqPost + 1) & N_CMD_MODMASK;
1548  ch = cmdQ.nSeqPost;
1549 
1550  pCmd = allocCmd(ch);
1551  }
1552  else
1553  {
1554  // No more RF command containers available -> set status and return error code
1555  Hwi_restore(key);

1557  }
1558

1574  if (Q_push(&cmdQ.pPend, pCmd) == false)

1577  Hwi_restore(key);
1578  return (RF_CmdHandle)RF_ALLOC_ERROR;
1579  }
1580

1582  pCmd->flags |= RF_CMD_ALLOC_FLAG;
1583  if (h->state.bYielded == true)
1584  {
1585  h->state.bYielded = false;
1586  bInitAvailRatCh = false;
1587  }
1588  Hwi_restore(key);
1589  return ch;
1590  }
1591 
1592  // Submit to pending command queue
1593  if (Q_push(&cmdQ.pPend, pCmd) == false)
1594  {
1595  cmdQ.nSeqPost = (N_CMD_MODMASK + cmdQ.nSeqPost) & N_CMD_MODMASK;
1596  Hwi_restore(key);
1597  return (RF_CmdHandle)RF_ALLOC_ERROR;
1598  }
1599 
1600  h->state.chLastPosted = ch;
1601  pCmd->flags |= RF_CMD_ALLOC_FLAG;
1602

1604  if (h->state.bYielded == true)

1614  bInitAvailRatCh = false;
1615  nActiveClients++;
1616  }
1617  Hwi_restore(key);

1631  {
1632  bRestart = true;
1633  Hwi_restore(key);
1634  }
1635  else if ((dispatchInUs = dispatchOpWhen(pCmd->pOp)))

1638  Clock_setTimeout(Clock_handle(&clkPowerUp), timeout);
1639  Clock_start(Clock_handle(&clkPowerUp));

1641  }
1642  else
1643  {
1644  Hwi_restore(key);
1645  Swi_or(Swi_handle(&swiFsm), Fsm_EventWakeup);

1647  }

1652 RF_EventMask RF_pendCmd(RF_Handle h, RF_CmdHandle ch, RF_EventMask bmEvent)
1653 {

1656  Assert_isTrue((ch >= 0), NULL);
1657 
1658 
1659  // Check whether command has finished, if not override command callback
1660  uint32_t key = Swi_disable();
1661 
1662  RF_Cmd* pCmd = getCmd(ch);
1663

1665  if (!pCmd)
1666  {
1667  Swi_restore(key);

1669  }
1670

1672  if (pCmd->flags & (RF_CMD_ABORTED_FLAG | RF_CMD_STOPPED_FLAG | RF_CMD_CANCELLED_FLAG))

1690  Semaphore_pend(Semaphore_handle(&h->state.semSync), BIOS_WAIT_FOREVER);
1691

1693  h->state.pCbSync = NULL;

1696  return h->state.unpendCause;

1774 static RF_Stat abortCmd(RF_Handle h, RF_CmdHandle ch, uint8_t type)
1775 {
1776  //Assert
1777  Assert_isTrue((h != NULL), NULL);
1778

1780  RF_Cmd* pCmdPend = Q_peek(&cmdQ.pPend);
1781  RF_CmdDirImm* pCmdDirImm = &cmdDirImm;
1782  bool bGraceful = type & RF_ABORT_CANCEL_SINGLE_CMD;
1783  RF_Stat status = RF_StatInvalidParamsError;
1784 
1785  // Check if current command
1786  uint32_t key = Hwi_disable();
1787  if (ch == RF_CMDHANDLE_FLUSH_ALL)
1788  {
1789  if (cmdQ.pCurrCmd)
1790  {
1791  ch = cmdQ.pCurrCmd->ch;
1792  }
1793  else
1794  {
1795  ch = cmdQ.nSeqDone + 1;
1796  }
1797  pCmdDirImm->flags = 0;
1798  }
1799  RF_Cmd* pCmd = getCmd(ch);
1800  if (pCmd && (pCmd == cmdQ.pCurrCmd))
1801  {
1802  // Clear all pending commands
1803  if (type & RF_ABORT_CANCEL_MULTIPLE_CMD)
1804  {
1805  pCmdPend = cmdQ.pPend;
1806  cmdQ.pPend = NULL;
1807  }
1808 
1809  // Flag that operation has been aborted.
1810  pCmd->flags |= (1 << bGraceful);
1811 
1812  uint32_t keySwi = Swi_disable();
1813 
1814  RFCDoorbellSendTo((bGraceful) ? CMDR_DIR_CMD(CMD_STOP) : CMDR_DIR_CMD(CMD_ABORT));
1815 
1816  Hwi_restore(key);
1817

1819  if (type & RF_ABORT_CANCEL_MULTIPLE_CMD)
1820  {
1821  while (pCmdPend)
1822  {
1823  pCmdPend->flags |= RF_CMD_CANCELLED_FLAG;
1824  Q_push(&cmdQ.pDone, Q_pop(&pCmdPend));
1825  }
1826  }
1827 
1828  Swi_restore(keySwi);
1829  status = RF_StatSuccess;
