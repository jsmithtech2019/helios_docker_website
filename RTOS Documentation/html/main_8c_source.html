117 /*******************************************************************************
118  * @fn AssertHandler
119  *
120  * @brief This is the Application's callback handler for asserts raised
121  * in the stack. When EXT_HAL_ASSERT is defined in the Stack
122  * project this function will be called when an assert is raised,
123  * and can be used to observe or trap a violation from expected
124  * behavior.
125  *
126  * As an example, for Heap allocation failures the Stack will raise
127  * HAL_ASSERT_CAUSE_OUT_OF_MEMORY as the assertCause and
128  * HAL_ASSERT_SUBCAUSE_NONE as the assertSubcause. An application
129  * developer could trap any malloc failure on the stack by calling
130  * HAL_ASSERT_SPINLOCK under the matching case.
131  *
132  * An application developer is encouraged to extend this function
133  * for use by their own application. To do this, add hal_assert.c
134  * to your project workspace, the path to hal_assert.h (this can
135  * be found on the stack side). Asserts are raised by including
136  * hal_assert.h and using macro HAL_ASSERT(cause) to raise an
137  * assert with argument assertCause. the assertSubcause may be
138  * optionally set by macro HAL_ASSERT_SET_SUBCAUSE(subCause) prior
139  * to asserting the cause it describes. More information is
140  * available in hal_assert.h.
141  *
142  * input parameters
143  *
144  * @param assertCause - Assert cause as defined in hal_assert.h.
145  * @param assertSubcause - Optional assert subcause (see hal_assert.h).
146  *
147  * output parameters
148  *
149  * @param None.
150  *
151  * @return None.
152  */
153 void AssertHandler(uint8 assertCause, uint8 assertSubcause)
154 {
155  // Open the display if the app has not already done so
156  if ( !dispHandle )
157  {
158  dispHandle = Display_open(Display_Type_LCD, NULL);
159  }
160 
161  Display_print0(dispHandle, 0, 0, ">>>STACK ASSERT");
162 
163  // check the assert cause
164  switch (assertCause)
165  {
166  case HAL_ASSERT_CAUSE_OUT_OF_MEMORY:
167  Display_print0(dispHandle, 0, 0, "***ERROR***");
168  Display_print0(dispHandle, 2, 0, ">> OUT OF MEMORY!");
169  break;
170 
171  case HAL_ASSERT_CAUSE_INTERNAL_ERROR:
172  // check the subcause
173  if (assertSubcause == HAL_ASSERT_SUBCAUSE_FW_INERNAL_ERROR)
174  {
175  Display_print0(dispHandle, 0, 0, "***ERROR***");
176  Display_print0(dispHandle, 2, 0, ">> INTERNAL FW ERROR!");
177  }
178  else
179  {
180  Display_print0(dispHandle, 0, 0, "***ERROR***");
181  Display_print0(dispHandle, 2, 0, ">> INTERNAL ERROR!");
182  }
183  break;
184 
185  case HAL_ASSERT_CAUSE_ICALL_ABORT:
186  Display_print0(dispHandle, 0, 0, "***ERROR***");
187  Display_print0(dispHandle, 2, 0, ">> ICALL ABORT!");
188  HAL_ASSERT_SPINLOCK;
189  break;
190 
191  default:
192  Display_print0(dispHandle, 0, 0, "***ERROR***");
193  Display_print0(dispHandle, 2, 0, ">> DEFAULT SPINLOCK!");
194  HAL_ASSERT_SPINLOCK;
195  }
196 
197  return;
198 }
