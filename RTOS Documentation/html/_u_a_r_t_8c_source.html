85  * ======== UART_init ========
86  */
87 void UART_init(void)
88 {
89  if (UART_count == -1) {
90  /* Call each driver's init function */
91  for (UART_count = 0; UART_config[UART_count].fxnTablePtr != NULL; UART_count++) {
92  UART_config[UART_count].fxnTablePtr->initFxn((UART_Handle)&(UART_config[UART_count]));
93  }
94  }
95 }

97 /*
98  * ======== UART_open ========
99  */
100 UART_Handle UART_open(unsigned int index, UART_Params *params)
101 {
102  UART_Handle handle;
103 
104  if (index >= UART_count) {
105  return (NULL);
106  }
107 
108  /* If params are NULL use defaults */
109  if (params == NULL) {
110  params = (UART_Params *) &UART_defaultParams;
111  }
112 
113  /* Get handle for this driver instance */
114  handle = (UART_Handle)&(UART_config[index]);
115 
116  return (handle->fxnTablePtr->openFxn(handle, params));
117 
118 }
