518 /*******************************************************************************
519  * @fn RegisterAssertCback
520  *
521  * @brief This routine registers the Application's assert handler.
522  *
523  * input parameters
524  *
525  * @param appAssertHandler - Application's assert handler.
526  *
527  * output parameters
528  *
529  * @param None.
530  *
531  * @return None.
532  */
533 void RegisterAssertCback( assertCback_t appAssertHandler )
534 {
535  appAssertCback = appAssertHandler;
536 
537 #ifdef EXT_HAL_ASSERT
538  // also set the Assert callback pointer used by halAssertHandlerExt
539  // Note: Normally, this pointer will be intialized by the stack, but in the
540  // event HAL_ASSERT is used by the Application, we initialize it
541  // directly here.
542  halAssertCback = appAssertHandler;
543 #endif // EXT_HAL_ASSERT
544 
545  return;
546 }
