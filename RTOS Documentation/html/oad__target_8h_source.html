158 /*
159  * OAD_IMG_VER() sets the least significant bit of the build version to signify
160  * which image region the image is built to run in. If the LSB is set to 1,
161  * then the Image B region is intended. Otherwise, it is built for the image A
162  * region. When that builds Image Header is sent OTA to the currently running
163  * Image, OAD_IMG_ID() is performed as a simple check to make sure that the
164  * current Image running and the Image to OTD are not both and Image A or
165  * Image B builds. OAD_VER_NUM() unmodifies the OAD image version number for
166  * uses such as writing to the LCD.
167  */
168 
169 // Macros to get Image ID (LSB) and Version Number
170 #define OAD_IMG_ID( ver ) ( (ver) & 0x01 )
171 #define OAD_VER_NUM( ver ) ( (ver) >> 0x01 )
172 
173 // Macro to set Image Version
174 #if defined (HAL_IMAGE_A)
175  // Clear LSB for Image A
176  #define OAD_IMG_VER( ver ) ( (uint16)( (ver) << 0x01 ) )
177 #elif defined (HAL_IMAGE_B)
178  // Set LSB for Image B
179  #define OAD_IMG_VER( ver ) ( (uint16)( ( (ver) << 0x01 ) | 0x01 ) )
180 #else
181  // Do nothing.
182  #define OAD_IMG_VER( ver ) (ver)
183 #endif // HAL_IMAGE_A
