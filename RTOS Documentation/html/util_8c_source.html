98 /*********************************************************************
99  * @fn Util_constructClock
100  *
101  * @brief Initialize a TIRTOS Clock instance.
102  *
103  * @param pClock - pointer to clock instance structure.
104  * @param clockCB - callback function upon clock expiration.
105  * @param clockDuration - longevity of clock timer in milliseconds
106  * @param clockPeriod - if set to a value other than 0, the first
107  * expiry is determined by clockDuration. All
108  * subsequent expiries use the clockPeriod value.
109  * @param startFlag - TRUE to start immediately, FALSE to wait.
110  * @param arg - argument passed to callback function.
111  *
112  * @return Clock_Handle - a handle to the clock instance.
113  */
114 Clock_Handle Util_constructClock(Clock_Struct *pClock,
115  Clock_FuncPtr clockCB,
116  uint32_t clockDuration,
117  uint32_t clockPeriod,
118  uint8_t startFlag,
119  UArg arg)
120 {
121  Clock_Params clockParams;
122 
123  // Convert clockDuration in milliseconds to ticks.
124  uint32_t clockTicks = clockDuration * (1000 / Clock_tickPeriod);
125 
126  // Setup parameters.
127  Clock_Params_init(&clockParams);
128 
129  // Setup argument.
130  clockParams.arg = arg;
131 
132  // If period is 0, this is a one-shot timer.
133  clockParams.period = clockPeriod * (1000 / Clock_tickPeriod);
134 
135  // Starts immediately after construction if true, otherwise wait for a call
136  // to start.
137  clockParams.startFlag = startFlag;
138 
139  // Initialize clock instance.
140  Clock_construct(pClock, clockCB, clockTicks, &clockParams);
141 
142  return Clock_handle(pClock);
143 }

172 void Util_restartClock(Clock_Struct *pClock, uint32_t clockTimeout)
173 {
174  uint32_t clockTicks;
175  Clock_Handle handle;
176 
177  handle = Clock_handle(pClock);
178 
179  if (Clock_isActive(handle))
180  {

184 
185  // Convert timeout in milliseconds to ticks.
186  clockTicks = clockTimeout * (1000 / Clock_tickPeriod);
187 
188  // Set the initial timeout
189  Clock_setTimeout(handle, clockTicks);
190 
191  // Start clock instance
192  Clock_start(handle);

230 /*********************************************************************
231  * @fn Util_rescheduleClock
232  *
233  * @brief Reschedule a clock by changing the timeout and period values.
234  *
235  * @param pClock - pointer to clock struct
236  * @param clockPeriod - longevity of clock timer in milliseconds
237  * @return none
238  */
239 void Util_rescheduleClock(Clock_Struct *pClock, uint32_t clockPeriod)
240 {
241  bool running;
242  uint32_t clockTicks;
243  Clock_Handle handle;
244 
245  handle = Clock_handle(pClock);
246  running = Clock_isActive(handle);
247 
248  if (running)
249  {
250  Clock_stop(handle);
251  }
252 
253  // Convert period in milliseconds to ticks.
254  clockTicks = clockPeriod * (1000 / Clock_tickPeriod);
255 
256  Clock_setTimeout(handle, clockTicks);
257  Clock_setPeriod(handle, clockTicks);
258 
259  if (running)
260  {
261  Clock_start(handle);
262  }
263 }

282 /*********************************************************************
283  * @fn Util_enqueueMsg
284  *
285  * @brief Creates a queue node and puts the node in RTOS queue.
286  *
287  * @param msgQueue - queue handle.
288  * @param sem - thread's event processing semaphore that queue is
289  * associated with.
290  * @param pMsg - pointer to message to be queued
291  *
292  * @return TRUE if message was queued, FALSE otherwise.
293  */
294 uint8_t Util_enqueueMsg(Queue_Handle msgQueue,
295 #ifdef ICALL_EVENTS
296  Event_Handle event,
297 #else
298  Semaphore_Handle sem,
299 #endif //ICALL_EVENTS
300  uint8_t *pMsg)
301 {
302  queueRec_t *pRec;
303 
304  // Allocated space for queue node.
305 #ifdef USE_ICALL
306  if ((pRec = ICall_malloc(sizeof(queueRec_t))))
307 #else
308  if ((pRec = (queueRec_t *)malloc(sizeof(queueRec_t))))
309 #endif
310  {
311  pRec->pData = pMsg;
312 
313  // This is an atomic operation
314  Queue_put(msgQueue, &pRec->_elem);
315 
316  // Wake up the application thread event handler.
317 #ifdef ICALL_EVENTS
318  if (event)
319  {
320  Event_post(event, UTIL_QUEUE_EVENT_ID);
321  }
322 #else
323  if (sem)
324  {
325  Semaphore_post(sem);
326  }
327 #endif //ICALL_EVENTS
328 
329  return TRUE;
330  }
331 
332  // Free the message.
333 #ifdef USE_ICALL
334  ICall_free(pMsg);
335 #else
336  free(pMsg);
337 #endif
338 
339  return FALSE;
340 }

373 /*********************************************************************
374  * @fn Util_convertBdAddr2Str
375  *
376  * @brief Convert Bluetooth address to string. Only needed when
377  * LCD display is used.
378  *
379  * @param pAddr - BD address
380  *
381  * @return BD address as a string
382  */
383 char *Util_convertBdAddr2Str(uint8_t *pAddr)
384 {
385  uint8_t charCnt;
386  char hex[] = "0123456789ABCDEF";
387  static char str[(2*B_ADDR_LEN)+3];
388  char *pStr = str;
389 
390  *pStr++ = '0';
391  *pStr++ = 'x';
392 
393  // Start from end of addr
394  pAddr += B_ADDR_LEN;
395 
396  for (charCnt = B_ADDR_LEN; charCnt > 0; charCnt--)
397  {
398  *pStr++ = hex[*--pAddr >> 4];
399  *pStr++ = hex[*pAddr & 0x0F];
400  }
401  pStr = NULL;
402 
403  return str;
404 }
