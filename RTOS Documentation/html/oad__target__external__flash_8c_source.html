164 /*******************************************************************************
165  * @fn OADTarget_getCurrentImageHeader
166  *
167  * @brief Get the header of the running image.
168  *
169  * @param pHdr - pointer to store running image header.
170  *
171  * @return TRUE if flash is successfully opened
172  */
173 uint8_t OADTarget_getCurrentImageHeader(img_hdr_t *pHdr)
174 {
175  uint8_t ret;
176 
177  ret = OADTarget_open();
178 
179  if (ret)
180  {
181  // N.B: it is assumed that OADTarget_storeImageHeader() has been called
182  // prior to this function.
183  uint32_t metaDataAddr;
184  ExtImageInfo_t tempHdr;
185 
186  if (imgInfo.imgType == EFL_OAD_IMG_TYPE_STACK
187  || imgInfo.imgType == EFL_OAD_IMG_TYPE_NP)
188  {
189  metaDataAddr = EFL_IMAGE_INFO_ADDR_BLE;
190  }
191  else // Assume imgInfo.imgType == EFL_OAD_IMG_TYPE_APP
192  {
193  metaDataAddr = EFL_IMAGE_INFO_ADDR_APP;
194  }
195 
196  ExtFlash_read(metaDataAddr, sizeof(ExtImageInfo_t), (uint8_t*)&tempHdr);
197  OADTarget_close();
198 
199  pHdr->len = tempHdr.len;
200 
201  // In case metadata does not exist, use 0x00
202  pHdr->ver = tempHdr.ver == 0xFF ? 0x00 : tempHdr.ver;
203  memcpy(pHdr->uid,tempHdr.uid,sizeof(tempHdr.uid));
204  pHdr->res[0] = HI_UINT16(tempHdr.addr);
205  pHdr->res[1] = LO_UINT16(tempHdr.addr);
206  pHdr->res[2] = tempHdr.imgType;
207  pHdr->res[3] = 0xFF;
208  }
209 
210  return ret;
211 }

213 /*******************************************************************************
214  * @fn OADTarget_validateNewImage
215  *
216  * @brief Determine if a new image should be downloaded or not based on
217  * target specific criteria.
218  *
219  * @param pValue - pointer to new Image header information
220  * @param pCur - pointer to contents of current image header
221  * @param blkTot - total number of blocks comprising new image.
222  *
223  * @return TRUE to begin OAD otherwise FALSE to reject the image.
224  */
225 uint8_t OADTarget_validateNewImage(uint8_t *pValue, img_hdr_t *pCur,
226  uint16_t blkTot)
227 {
228  img_hdr_t *pNew;
229  uint32_t addr;
230  uint8_t imgType;
231  uint8_t valid;
232 
233  pNew = (img_hdr_t *)pValue;
234  addr = BUILD_UINT16(pNew->res[0],pNew->res[1]) * EFL_OAD_ADDR_RESOLUTION;
235  imgType = pNew->res[2];
236  valid = FALSE;
237 
238  // Check if number of blocks make sense
239  if (blkTot > MAX_BLOCKS || blkTot == 0)
240  {
241  return FALSE;
242  }
243 
244  //Note that network processor images can/will take up the entire internal
245  //flash space
246  if (imgType != EFL_OAD_IMG_TYPE_NP
247  && (addr < APP_IMAGE_START || addr > BOOT_LOADER_START))
248  {
249  return FALSE;
250  }
251 
252  // Image type 'APP' must start at 0x1000
253  if (addr != APP_IMAGE_START && imgType == EFL_OAD_IMG_TYPE_APP)
254  {
255  return FALSE;
256  }
257 
258  // Check if current header is invalid
259  if (pCur->ver == 0xFFFF || pCur->ver == 0x0000)
260  {
261  // Accept the image.
262  return TRUE;
263  }
264 
265  // By default, accept an image if version is 0.
266  if (pNew->ver == 0)
267  {
268  valid = TRUE;
269  }
270 
271  // If not already validated, check if new image is a later version than the
272  // current image.
273  if (!valid)
274  {
275  valid = pNew->ver > pCur->ver;
276  }
277 
278  return valid;
279 }

281 /*******************************************************************************
282  * @fn OADTarget_storeImageHeader
283  *
284  * @brief Store the image header of the new image
285  *
286  * @param pValue - pointer to the new image header
287  *
288  * @return none
289  */
290 void OADTarget_storeImageHeader(uint8_t *pValue)
291 {
292  img_hdr_t* pNew;
293 
294  pNew = (img_hdr_t*)(pValue +4);
295 
296  // Storage image header (written to external flash before reboot)
297  imgInfo.crc[0] = BUILD_UINT16(pValue[0],pValue[1]);
298  imgInfo.crc[1] = BUILD_UINT16(pValue[2],pValue[3]);
299  imgInfo.addr = BUILD_UINT16(pNew->res[0],pNew->res[1]);
300  imgInfo.ver = pNew->ver;
301  imgInfo.len = pNew->len;
302  memcpy(imgInfo.uid,pNew->uid,sizeof(imgInfo.uid));
303  imgInfo.imgType = pNew->res[2];
304 }

306 /*******************************************************************************
307  * @fn OADTarget_imageAddress
308  *
309  * @brief Get the address to store the new image
310  *
311  * @param pValue - pointer to the new image header
312  *
313  * @param pHdr - pointer to the current image header
314  *
315  * @return address
316  */
317 uint32_t OADTarget_imageAddress(uint8_t *pValue)
318 {
319  img_hdr_t *pNew;
320  uint32_t extAddr;
321 
322  pNew = (img_hdr_t *)pValue;
323 
324  switch (pNew->res[2])
325  {
326  // Application start at the beginning of external flash.
327  case EFL_OAD_IMG_TYPE_APP:
328  extAddr = EFL_ADDR_IMAGE_APP;
329  break;
330 
331  // All other images are placed into the next available image slot.
332  default:
333  extAddr = EFL_ADDR_IMAGE_BLE;
334  break;
335  }
336 
337  return extAddr;
338 }
