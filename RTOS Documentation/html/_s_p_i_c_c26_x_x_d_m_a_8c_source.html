155 /*
156  * Ensure safe setting of the standby disallow constraint.
157  */
158 static inline void threadSafeConstraintSet(uint32_t txBufAddr, SPICC26XXDMA_Object *object) {
159  unsigned int key;
160 
161  /* Disable interrupts */
162  key = Hwi_disable();
163 
164  if (!object->spiPowerConstraint) {
165  /* Ensure flash is available if TX buffer is in flash. Flash starts with 0x0..*/
166  if (((txBufAddr & 0xF0000000) == 0x0) && (txBufAddr)) {
167  Power_setConstraint(PowerCC26XX_NEED_FLASH_IN_IDLE);
168  }
169  /* Set constraints to guarantee operation */
170  Power_setConstraint(PowerCC26XX_SB_DISALLOW);
171  object->spiPowerConstraint = true;
172  }
173 
174  /* Re-enable interrupts */
175  Hwi_restore(key);
176 }
177 
178 /*
179  * Ensure safe releasing of the standby disallow constraint.
180  */
181 static inline void threadSafeConstraintRelease(uint32_t txBufAddr, SPICC26XXDMA_Object *object) {
182  unsigned int key;
183  /* Disable interrupts */
184  key = Hwi_disable();
185 
186  if (object->spiPowerConstraint) {
187  /* Release need flash if buffer was in flash. */
188  if (((txBufAddr & 0xF0000000) == 0x0) && (txBufAddr)) {
189  Power_releaseConstraint(PowerCC26XX_NEED_FLASH_IN_IDLE);
190  }
191  /* Release standby constraint since operation is done. */
192  Power_releaseConstraint(PowerCC26XX_SB_DISALLOW);
193  object->spiPowerConstraint = false;
194  }
195 
196  /* Re-enable interrupts */
197  Hwi_restore(key);
198 }

251

253  key = Hwi_disable();
254  object->isOpen = false;
255  Hwi_restore(key);
256 
257  Log_print1(Diags_USER1, "SPI:(%p) closed", hwAttrs->baseAddr);

280 int SPICC26XXDMA_control(SPI_Handle handle, unsigned int cmd, void *arg)
281 {
282  SPICC26XXDMA_Object *object;
283  SPICC26XXDMA_HWAttrs const *hwAttrs;
284  PIN_Config pinConfig;
285  PIN_Id pinId;
286 
287  /* Get the pointer to the object and hwAttr */
288  hwAttrs = handle->hwAttrs;
289  object = handle->object;
290 
291  /* Initialize return value*/
292  int ret = SPI_STATUS_ERROR;
293 
294  /* Perform command */
295  switch(cmd) {
296  case SPICC26XXDMA_CMD_RETURN_PARTIAL_ENABLE:
297  /* Enable RETURN_PARTIAL if slave mode is enabled */
298 
299  if(object->mode == SPI_SLAVE){
300  object->returnPartial = true;
301  PIN_setInterrupt(object->pinHandle, object->csnPin | PIN_IRQ_POSEDGE);
302  ret = SPI_STATUS_SUCCESS;
303  }
304  else{
305  /* Partial return not available in master mode. */
306  ret = SPI_STATUS_ERROR;
307  }
308  break;
309 
310  case SPICC26XXDMA_CMD_RETURN_PARTIAL_DISABLE:
311  /* Disable RETURN_PARTIAL */
312  object->returnPartial = false;
313  PIN_setInterrupt(object->pinHandle, object->csnPin | PIN_IRQ_DIS);
314  ret = SPI_STATUS_SUCCESS;
315  break;
316 
317  case SPICC26XXDMA_CMD_SET_CSN_PIN:
318  pinId = ((*(PIN_Id *) arg));
319 
320  /* Configure CSN pin and remap PIN_ID to new CSN pin specified by arg */
321  if (object->mode == SPI_SLAVE) {
322  pinConfig = PIN_INPUT_EN | PIN_PULLUP | pinId;
323  }
324  else {
325  pinConfig = PIN_GPIO_OUTPUT_EN | PIN_GPIO_HIGH | PIN_PUSHPULL | PIN_INPUT_DIS | PIN_DRVSTR_MED | pinId;
326  }
327 
328  if (pinId != PIN_UNASSIGNED) {
329  /* Attempt to add the new pin */
330  if (PIN_add(object->pinHandle, pinConfig) == PIN_SUCCESS) {
331  /* Configure pin mux */
332  PINCC26XX_setMux(object->pinHandle, pinId, (hwAttrs->baseAddr == SSI0_BASE ? IOC_PORT_MCU_SSI0_FSS : IOC_PORT_MCU_SSI1_FSS));
333 
334  /* Remove old pin and revert to default setting specified in the board file */
335  PIN_remove(object->pinHandle, object->csnPin);
336 
337  /* Keep track of current CSN pin */
338  object->csnPin = pinId;
339 
340  /* Add csn IRQ if return partial is enabled */
341  if((object->mode == SPI_SLAVE) && (object->returnPartial == true)){
342  PIN_setInterrupt(object->pinHandle, object->csnPin | PIN_IRQ_POSEDGE);
343  }
344 
345  /* Set return value to indicate success */
346  ret = SPI_STATUS_SUCCESS;
347  }
348  }
349  else {
350  /* We want to use software ctrl CSN. Hence, undo any prior hardware CSN pin muxing */
351  /* Remove old pin and revert to default setting specified in the board file (implicitly sets IO muxing to GPIO mode) */
352  PIN_remove(object->pinHandle, object->csnPin);
353 
354  /* Keep track of current CSN pin */
355  object->csnPin = pinId;
356 
357  /* Set return value to indicate success */
358  ret = SPI_STATUS_SUCCESS;
359  }
360  break;
361 
362 #ifdef SPICC26XXDMA_WAKEUP_ENABLED
363  case SPICC26XXDMA_CMD_SET_CSN_WAKEUP:
364  /* Set wakeup callback function no matter what, NULL signals that the wakeup feature is disabled */
365  object->wakeupCallbackFxn = *(SPICC26XXDMA_CallbackFxn)arg;
366  ret = SPI_STATUS_SUCCESS;
367  break;
368 #endif
369  default:
370  /* This command is not defined */
371  ret = SPI_STATUS_UNDEFINEDCMD;
372  break;
373  }
374 
375  return (ret);
376 }

378 /*
379  * ======== SPICC26XXDMA_configDMA ========
380  * This functions configures the transmit and receive DMA channels for a given
381  * SPI_Handle and SPI_Transaction
382  *
383  * @pre Function assumes that the handle and transaction is not NULL
384  */
385 static void SPICC26XXDMA_configDMA(SPI_Handle handle, SPI_Transaction *transaction)
386 {
387  SPICC26XXDMA_Object *object;
388  SPICC26XXDMA_HWAttrs const *hwAttrs;
389  volatile tDMAControlTable *dmaControlTableEntry;
390  uint16_t numberOfBytes;
391 
392  /* Get the pointer to the object and hwAttrs */
393  object = handle->object;
394  hwAttrs = handle->hwAttrs;
395 
396  /* Calculate the number of bytes for the transfer */
397  numberOfBytes = ((uint16_t)(transaction->count) << (object->frameSize));
398 
399  /* Setup RX side */
400  /* Set pointer to Rx control table entry */
401  dmaControlTableEntry = (hwAttrs->baseAddr == SSI0_BASE ? &dmaSpi0RxControlTableEntry : &dmaSpi1RxControlTableEntry);
402  if (transaction->rxBuf) {
403  dmaControlTableEntry->ui32Control = dmaRxConfig[object->frameSize];
404  dmaControlTableEntry->pvDstEndAddr = (void *)((uint32_t)(transaction->rxBuf) + numberOfBytes - 1);
405  }
406  else {
407  dmaControlTableEntry->ui32Control = dmaNullConfig[object->frameSize];
408  dmaControlTableEntry->pvDstEndAddr = (void *) &(object->scratchBuf);
409  }
410  dmaControlTableEntry->pvSrcEndAddr = (void *)(hwAttrs->baseAddr + SSI_O_DR);
411  dmaControlTableEntry->ui32Control |= UDMACC26XX_SET_TRANSFER_SIZE((uint16_t)transaction->count);
412 
413  /* Setup TX side */
414  /* Set pointer to Tx control table entry */
415  dmaControlTableEntry = (hwAttrs->baseAddr == SSI0_BASE ? &dmaSpi0TxControlTableEntry : &dmaSpi1TxControlTableEntry);
416  if (transaction->txBuf) {
417  dmaControlTableEntry->ui32Control = dmaTxConfig[object->frameSize];
418  dmaControlTableEntry->pvSrcEndAddr = (void *)((uint32_t)(transaction->txBuf) + numberOfBytes - 1);
419  }
420  else {
421  object->scratchBuf = hwAttrs->defaultTxBufValue;
422  dmaControlTableEntry->ui32Control = dmaNullConfig[object->frameSize];
423  dmaControlTableEntry->pvSrcEndAddr = (void *) &(object->scratchBuf);
424  }
425  dmaControlTableEntry->pvDstEndAddr = (void *)(hwAttrs->baseAddr + SSI_O_DR);
426  dmaControlTableEntry->ui32Control |= UDMACC26XX_SET_TRANSFER_SIZE((uint16_t)transaction->count);
427 
428  /* Enable the channels */
429  UDMACC26XX_channelEnable(object->udmaHandle, (hwAttrs->rxChannelBitMask) | (hwAttrs->txChannelBitMask));
430 
431  /* Enable the required DMA channels in the SPI module to start the transaction */
432  SSIDMAEnable(hwAttrs->baseAddr, SSI_DMA_TX | SSI_DMA_RX);
433 
434  Log_print1(Diags_USER1,"SPI:(%p) DMA transfer enabled", hwAttrs->baseAddr);
435 
436  Log_print5(Diags_USER2,"SPI:(%p) DMA transaction: %p, "
437  "rxBuf: %p; txBuf: %p; Count: %d",
438  hwAttrs->baseAddr,
439  (UArg)transaction,
440  (UArg)transaction->rxBuf,
441  (UArg)transaction->txBuf,
442  (UArg)transaction->count);
443 }

445 /*
446  * ======== SPICC26XXDMA_hwiFxn ========
447  * HWI ISR for the SPI when we use the UDMA
448  */
449 static void SPICC26XXDMA_hwiFxn (UArg arg) {
450  SPICC26XXDMA_Object *object;
451  SPICC26XXDMA_HWAttrs const *hwAttrs;
452  uint32_t intStatus;
453 
454  /* Get the pointer to the object and hwAttrs */
455  object = ((SPI_Handle)arg)->object;
456  hwAttrs = ((SPI_Handle)arg)->hwAttrs;
457 
458  Log_print1(Diags_USER2, "SPI:(%p) hwi interrupt context start", hwAttrs->baseAddr);
459 
460  /* Get the interrupt status of the SPI controller
461  */
462  intStatus = SSIIntStatus(hwAttrs->baseAddr, true);
463  SSIIntClear(hwAttrs->baseAddr, intStatus);
464 
465  /* Error handling:
466  * Overrun in the RX Fifo -> at least one sample in the shift
467  * register has been discarded */
468  if (intStatus & SSI_RXOR) {
469  /* disable the interrupt */
470  SSIIntDisable(hwAttrs->baseAddr, SSI_RXOR);
471 
472  /* If the RX overrun occurred during a transfer */
473  if (object->currentTransaction) {
474  /* Then cancel the ongoing transfer */
475  SPICC26XXDMA_transferCancel((SPI_Handle)arg);
476  }
477  else {
478  /* Otherwise disable the SPI and DMA modules and flush FIFOs */
479  SSIDisable(hwAttrs->baseAddr);
480 
481  /* Disable SPI TX/RX DMA and clear DMA done interrupt just in case it finished */
482  SSIDMADisable(hwAttrs->baseAddr, SSI_DMA_TX | SSI_DMA_RX);
483  UDMACC26XX_clearInterrupt(object->udmaHandle, (hwAttrs->rxChannelBitMask) | (hwAttrs->txChannelBitMask));
484 
485  /* Clear out the FIFO by resetting SPI module and re-initting */
486  SPICC26XXDMA_flushFifos((SPI_Handle)arg);
487  }
488  Log_print1(Diags_USER1, "RX FIFO overrun occurred in SPI: (%p) !\n", hwAttrs->baseAddr);
489  }
490  else {
491  /*
492  * Determine if the TX DMA channel has completed... In SPI slave mode
493  * this interrupt may occur immediately (without the RX DMA channel).
494  *
495  * All transfers will set up both TX and RX DMA channels and both will finish.
496  * Even if the transaction->rxBuf == NULL, it will setup a dummy RX transfer to
497  * a scratch memory location which is then discarded.
498  */
499  if (UDMACC26XX_channelDone(object->udmaHandle, hwAttrs->txChannelBitMask)) {
500  /* Disable SPI TX DMA and clear DMA done interrupt. */
501  SSIDMADisable(hwAttrs->baseAddr, SSI_DMA_TX);
502  UDMACC26XX_clearInterrupt(object->udmaHandle, hwAttrs->txChannelBitMask);
503  }
504 
505  /*
506  * Determine if the RX DMA channel has completed... In slave mode this interrupt
507  * occurrence depends on when the SPI master starts sending data.
508  */
509  if (UDMACC26XX_channelDone(object->udmaHandle, hwAttrs->rxChannelBitMask)) {
510  /* Disable SPI RX DMA and clear DMA done interrupt. */
511  SSIDMADisable(hwAttrs->baseAddr, SSI_DMA_RX);
512  UDMACC26XX_clearInterrupt(object->udmaHandle, hwAttrs->rxChannelBitMask);
513 
514  /* Post SWI to handle remaining clean up and invocation of callback */
515  Swi_post(Swi_handle(&(object->swi)));
516  }
517  }
518 
519  Log_print1(Diags_USER2, "SPI:(%p) hwi interrupt context end",
520  hwAttrs->baseAddr);
521 }

523 /*
524  * ======== SPICC26XXDMA_swiFxn ========
525  * SWI ISR for the SPI when we use the UDMA
526  * Determine if the RX DMA channel has completed... In slave mode this interrupt
527  * occurrence depends on when the SPI master starts sending data.
528  */
529 static void SPICC26XXDMA_swiFxn (UArg arg0, UArg arg1) {
530  SPI_Transaction *msg;
531  SPICC26XXDMA_Object *object;
532 
533  /* Get the pointer to the object and hwAttrs */
534  object = ((SPI_Handle)arg0)->object;
535 
536  Log_print1(Diags_USER2, "SPI:(%p) swi interrupt context start",
537  ((SPICC26XXDMA_HWAttrs const *)(((SPI_Handle)arg0)->hwAttrs))->baseAddr);
538 
539  /* Clean up SPI and DMA */
540  /* Transaction is complete */
541  object->currentTransaction->status = SPI_TRANSFER_COMPLETED;
542 
543  /* Use a temporary transaction pointer in case the callback function
544  * attempts to perform another SPI_transfer call
545  */
546  msg = object->currentTransaction;
547 
548  Log_print2(Diags_USER1,"SPI:(%p) DMA transaction: %p complete",
549  ((SPICC26XXDMA_HWAttrs const *)(((SPI_Handle)arg0)->hwAttrs))->baseAddr, (UArg)msg);
550 
551  /* Release constraint since transaction is done */
552  threadSafeConstraintRelease((uint32_t)(object->currentTransaction->txBuf), object);
553 
554  /* Indicate we are done with this transfer */
555  object->currentTransaction = NULL;
556 
557  /* Perform callback */
558  object->transferCallbackFxn((SPI_Handle)arg0, msg);
559 
560  Log_print1(Diags_USER2, "SPI:(%p) swi interrupt context end",
561  ((SPICC26XXDMA_HWAttrs const *)(((SPI_Handle)arg0)->hwAttrs))->baseAddr);
562 }

572 /*
573  * ======== SPICC26XXDMA_flushTxFifo ========
574  */
575 void SPICC26XXDMA_flushFifos(SPI_Handle handle) {
576 
577  /* Locals */
578  SPICC26XXDMA_HWAttrs const *hwAttrs;
579 
580  /* Get the pointer to the hwAttrs */
581  hwAttrs = handle->hwAttrs;
582 
583  /* Flush RX FIFO */
584  while(HWREG(hwAttrs->baseAddr + SSI_O_SR) & SSI_RX_NOT_EMPTY) {
585  /* Read element from RX FIFO and discard */
586  HWREG(hwAttrs->baseAddr + SSI_O_DR);
587  }
588 
589  /* Enable TESTFIFO mode */
590  HWREG(hwAttrs->baseAddr + SSI_O_TCR) = SSI_TCR_TESTFIFO_ENABLE;
591 
592  /* Flush TX FIFO */
593  while(!(HWREG(hwAttrs->baseAddr + SSI_O_SR) & SSI_TX_EMPTY)) {
594  /* Read element from TX FIFO and discard */
595  HWREG(hwAttrs->baseAddr + SSI_O_TDR);
596  }
597 
598  /* Disable TESTFIFO mode */
599  HWREG(hwAttrs->baseAddr + SSI_O_TCR) = SSI_TCR_TESTFIFO_DISABLE;
600 }

812 bool SPICC26XXDMA_transfer(SPI_Handle handle, SPI_Transaction *transaction)
813 {
814  unsigned int key;
815  SPICC26XXDMA_Object *object;
816  SPICC26XXDMA_HWAttrs const *hwAttrs;
817 
818  /* Get the pointer to the object and hwAttr*/
819  object = handle->object;
820  hwAttrs = handle->hwAttrs;
821 
822  /* This is a limitation by the uDMA controller */
823  Assert_isTrue(transaction->count <= 1024, NULL);
824 
825  if (transaction->count == 0) {
826  return (false);
827  }
828 
829  /* Make sure that the buffers are aligned properly */
830  if (object->frameSize == SPICC26XXDMA_16bit) {
831  Assert_isTrue(!((unsigned long)transaction->txBuf & 0x1), NULL);
832  Assert_isTrue(!((unsigned long)transaction->rxBuf & 0x1), NULL);
833  }
834 
835  /* Disable preemption while checking if a transfer is in progress */
836  key = Hwi_disable();
837  if (object->currentTransaction) {
838  Hwi_restore(key);
839 
840  Log_error1("SPI:(%p) transaction still in progress",
841  ((SPICC26XXDMA_HWAttrs const *)(handle->hwAttrs))->baseAddr);
842 
843  /* Flag that the transfer failed to start */
844  transaction->status = SPI_TRANSFER_FAILED;
845 
846  /* Transfer is in progress */
847  return (false);
848  }
849 
850  /* Make sure to flag that a transaction is now active */
851  transaction->status = SPI_TRANSFER_STARTED;
852  object->currentTransaction = transaction;
853 
854  Hwi_restore(key);
855 
856  /* Set constraints to guarantee transaction */
857  threadSafeConstraintSet((uint32_t)(transaction->txBuf), object);
858 
859  /* Enable the SPI module */
860  SSIEnable(hwAttrs->baseAddr);
861 
862  /* Setup DMA transfer. */
863  SPICC26XXDMA_configDMA(handle, transaction);
864 
865  /* Enable the RX overrun interrupt in the SSI module */
866  SSIIntEnable(hwAttrs->baseAddr, SSI_RXOR);
867 
868  if (object->transferMode == SPI_MODE_BLOCKING) {
869  Log_print1(Diags_USER1, "SPI:(%p) transfer pending on transferComplete "
870  "semaphore",
871  ((SPICC26XXDMA_HWAttrs const *)(handle->hwAttrs))->baseAddr);
872 
873  if (!Semaphore_pend(Semaphore_handle(&(object->transferComplete)), object->transferTimeout)) {
874  /* Cancel the transfer, if we experience a timeout */
875  SPICC26XXDMA_transferCancel(handle);
876  /*
877  * SPICC26XXDMA_transferCancel peforms a callback which posts a
878  * transferComplete semaphore. This call consumes this extra post.
879  */
880  Semaphore_pend(Semaphore_handle(&(object->transferComplete)),
881  BIOS_NO_WAIT);
882  return (false);
883  }
884  }
885  return (true);
886 }

897 void SPICC26XXDMA_transferCancel(SPI_Handle handle) {
898  SPICC26XXDMA_Object *object;
899  SPI_Transaction *msg;
900  SPICC26XXDMA_HWAttrs const *hwAttrs;
901  volatile tDMAControlTable *dmaControlTableEntry;
902  unsigned int key;
903 
904  /* Get the pointer to the object and hwAttrs */
905  object = handle->object;
906  hwAttrs = handle->hwAttrs;
907 
908  /* Check if a transfer is in progress */
909  key = Hwi_disable();
910 
911  /* Check if there is an active transaction */
912  if(!(object->currentTransaction)) {
913  Hwi_restore(key);
914  return;
915  }
916  Hwi_restore(key);
917 
918  /* Disable the SPI module */
919  SSIDisable(hwAttrs->baseAddr);
920 
921  /* Disable SPI TX/RX DMA and clear DMA done interrupt just in case it finished */
922  SSIDMADisable(hwAttrs->baseAddr, SSI_DMA_TX | SSI_DMA_RX);
923  UDMACC26XX_clearInterrupt(object->udmaHandle, (hwAttrs->rxChannelBitMask) | (hwAttrs->txChannelBitMask));
924 
925  /* Disable and clear any pending interrupts */
926  SSIIntDisable(hwAttrs->baseAddr, SSI_RXOR);
927  SSIIntClear(hwAttrs->baseAddr, SSI_RXOR);
928 
929  /* Flush the FIFOs and re-initialize module */
930  SPICC26XXDMA_flushFifos(handle);
931 
932  /* Release constraint since transaction is done */
933  threadSafeConstraintRelease((uint32_t)(object->currentTransaction->txBuf), object);
934 
935  /* Mark the transaction as failed if we didn't end up here due to a CSN deassertion */
936  if (object->currentTransaction->status != SPI_TRANSFER_CSN_DEASSERT) {
937  object->currentTransaction->status = SPI_TRANSFER_FAILED;
938  }
939 
940  /* Disable the UDMA channels */
941  UDMACC26XX_channelDisable(object->udmaHandle, (hwAttrs->rxChannelBitMask) | (hwAttrs->txChannelBitMask));
942 
943  /* Update the SPI_Transaction.count parameter */
944  /* rxChannel always finishes after txChannel so remaining bytes of the rxChannel is used to update count */
945  dmaControlTableEntry = (hwAttrs->baseAddr == SSI0_BASE ? &dmaSpi0RxControlTableEntry : &dmaSpi1RxControlTableEntry);
946  object->currentTransaction->count -= UDMACC26XX_GET_TRANSFER_SIZE(dmaControlTableEntry->ui32Control);
947 
948  /* Use a temporary transaction pointer in case the callback function
949  * attempts to perform another SPI_transfer call
950  */
951  msg = object->currentTransaction;
952 
953  /* Indicate we are done with this transfer */
954  object->currentTransaction = NULL;
955 
956  Log_print2(Diags_USER1,"SPI:(%p) DMA transaction: %p cancelled",
957  hwAttrs->baseAddr, (UArg)msg);
958 
959  /* Perform callback */
960  object->transferCallbackFxn(handle, msg);
961 
962  /* Transaction was successfully canceled */
963  return;
964 }
