133 /*
134  * Ensure safe setting of the standby disallow constraint.
135  */
136 static inline void threadSafeStdbyDisSet(volatile bool *pConstraint) {
137  unsigned int key;
138 
139  /* Disable interrupts */
140  key = Hwi_disable();
141 
142  /* Only act if the current constraint is not previously set */
143  if (!*pConstraint) {
144  /* Set constraints to guarantee operation */
145  Power_setConstraint(PowerCC26XX_SB_DISALLOW);
146  *pConstraint = true;
147  }
148 
149  /* Re-enable interrupts */
150  Hwi_restore(key);
151 }

153 /*
154  * Ensure safe releasing of the standby disallow constraint.
155  */
156 static inline void threadSafeStdbyDisRelease(volatile bool *pConstraint) {
157  unsigned int key;
158 
159  /* Disable interrupts */
160  key = Hwi_disable();
161 
162  if (*pConstraint) {
163  /* release constraint since operation is done */
164  Power_releaseConstraint(PowerCC26XX_SB_DISALLOW);
165  *pConstraint = false;
166  }
167 
168  /* Re-enable interrupts */
169  Hwi_restore(key);
170 }

200 /*
201  * ======== writeData ========
202  * Write and process data to the UART.
203  */
204 static int32_t writeData(UART_Handle handle, int32_t size)
205 {
206  UARTCC26XX_Object *object;
207  UARTCC26XX_HWAttrsV2 const *hwAttrs;
208 
209  /* Get the pointer to the object and hwAttrs */
210  object = handle->object;
211  hwAttrs = handle->hwAttrs;
212 
213  /* Send characters until FIFO is full or done. */
214  while (size) {
215  /* Send the next character and increment counts. */
216  if (!UARTCharPutNonBlocking(hwAttrs->baseAddr, *(unsigned char *)(object->writeBuf))) {
217  /* Character was not sent */
218  break;
219  }
220  Log_print2(Diags_USER2, "UART:(%p) Wrote character 0x%x",
221  hwAttrs->baseAddr, *(unsigned char *)object->writeBuf);
222  object->writeBuf = (unsigned char *)object->writeBuf + 1;
223  size--;
224  object->writeCount++;
225  }
226  return (size);
227 }

229 /*
230  * ======== readData ========
231  * Read and process data from the UART.
232  * @param(size) number of bytes to be read
233  */
234 static int32_t readData(UART_Handle handle, int32_t size)
235 {
236  int32_t readIn;
237  UARTCC26XX_Object *object;
238  UARTCC26XX_HWAttrsV2 const *hwAttrs;
239 
240  /* Get the pointer to the object and hwAttrs */
241  object = handle->object;
242  hwAttrs = handle->hwAttrs;
243 
244  /* Receive chars until empty or done. */
245  while (size && (readIn = (int32_t)UARTCharGetNonBlocking(hwAttrs->baseAddr)) != -1) {
246 
247  Log_print2(Diags_USER2, "UART:(%p) Read character 0x%x",
248  hwAttrs->baseAddr, (uint8_t)readIn);
249 
250  /* Update status. */
251  *((unsigned char *)object->readBuf) = (uint8_t)readIn;
252  object->readBuf = (unsigned char *)object->readBuf + 1;
253  object->readCount++;
254  size--;
255  }
256  return (size);
257 }

288 /*
289  * ======== startTxFifoEmptyClk ========
290  * Last write to TX FIFO is done, but not shifted out yet. Start a clock
291  * which will trigger when the TX FIFO should be empty.
292  *
293  * @param handle The UART_Handle for ongoing write.
294  * @param numOfDataInFifo The number of data present in FIFO after last write
295  */
296 static void startTxFifoEmptyClk(UART_Handle handle, unsigned int numOfDataInFifo)
297 {
298  UARTCC26XX_Object *object;
299 
300  /* Get the pointer to the object and hwAttrs */
301  object = handle->object;
302 
303  /* Ensure that the clock is stopped so we can set a new timeout */
304  Clock_stop((Clock_Handle) &(object->txFifoEmptyClk));
305 
306  /* No more to write, but data is not shifted out properly yet.
307  * 1. Compute appropriate wait time for FIFO to empty out
308  * - 1 bit for start bit
309  * - 5+(object->dataLength) for total data length
310  * - +1 to "map" from stopBits to actual number of bits
311  * - 1000000 so we get 1 us resolution
312  * - 100 (100us) for margin
313  */
314  unsigned int writeTimeoutUs = (numOfDataInFifo*(1+5+(object->dataLength)+(object->stopBits+1))*1000000)/object->baudRate + 100;
315  /* 2. Configure clock object to trigger when FIFO is empty
316  * - +1 in case clock module due to tick in less than one Clock_tickPeriod
317  * - UNIT_DIV_ROUNDUP to avoid fractional part being truncated during division
318  */
319 
320  Clock_setTimeout((Clock_Handle) &(object->txFifoEmptyClk),
321  (1 + UNIT_DIV_ROUNDUP(writeTimeoutUs, Clock_tickPeriod)));
322  Clock_start((Clock_Handle) &(object->txFifoEmptyClk));
323 }

325 /*
326  * ======== writeFinishedDoCallback ========
327  * Write finished - make callback
328  *
329  * This function is called when the txFifoEmptyClk times out. The TX FIFO
330  * should now be empty and all bytes have been transmitted. The TX will be
331  * turned off, TX interrupt is disabled and standby is allowed again.
332  *
333  * @param(handle) The UART_Handle for ongoing write.
334  */
335 static void writeFinishedDoCallback(UART_Handle handle)
336 {
337  UARTCC26XX_Object *object;
338  UARTCC26XX_HWAttrsV2 const *hwAttrs;
339 
340  /* Get the pointer to the object and hwAttrs */
341  object = handle->object;
342  hwAttrs = handle->hwAttrs;
343 
344  /* Stop the txFifoEmpty clock */
345  Clock_stop((Clock_Handle) &(object->txFifoEmptyClk));
346 
347  /* Function verifies that the FIFO is empty via BUSY flag */
348  /* If not yet ready start the periodic timer and wait another period*/
349  /* Return.. */
350  if(UARTBusy(hwAttrs->baseAddr)){
351  /* The UART is still busy.
352  * Wait 500 us before checking again or 1 tick period if the
353  * Clock_tickPeriod is larger than 500 us.
354  */
355  Clock_setTimeout((Clock_Handle) &(object->txFifoEmptyClk), MAX((500/Clock_tickPeriod),1));
356  Clock_start((Clock_Handle) &(object->txFifoEmptyClk));
357  return;
358  }
359 
360  /* Release constraint since transaction is done */
361  threadSafeStdbyDisRelease(&uartTxPowerConstraint);
362 
363  /* Disable TX interrupt */
364  UARTIntDisable(hwAttrs->baseAddr, UART_INT_TX);
365 
366  /* Disable TX */
367  HWREG(UART0_BASE + UART_O_CTL) &= ~(UART_CTL_TXE);
368 
369  /* Reset the write buffer so we can pass it back */
370  object->writeBuf = (unsigned char *)object->writeBuf - object->writeCount;
371 
372  /* Make callback */
373  object->writeCallback(handle, (uint8_t*)object->writeBuf,
374  object->writeCount);
375  Log_print2(Diags_USER1, "UART:(%p) Write finished, %d bytes written",
376  hwAttrs->baseAddr, object->writeCount);
377 }

379 /*
380  * ======== writeTxFifoFlush ========
381  * Write cancelled or timed out, the TX FIFO must be flushed out.
382  *
383  * This function is called either from writeCancel or when a blocking write
384  * has timed out. The HW does not support a simple API for flushing the TX FIFO
385  * so a workaround is done in SW.
386  *
387  * @pre The TX FIFO empty clock must have been started in blocking mode.
388  *
389  * @param object Pointer to UART object
390  * @param hwAttrs Pointer to UART hwAttrs
391  */
392 static void writeTxFifoFlush(UARTCC26XX_Object *object, UARTCC26XX_HWAttrsV2 const *hwAttrs)
393 {
394  /*It is not possible to flush the TX FIFO with simple write to HW, doing workaround:
395  * 0. Disable TX interrupt
396  */
397  UARTIntDisable(hwAttrs->baseAddr, UART_INT_TX);
398  /* 1. Ensure TX IO will stay high when connected to GPIO */
399  PIN_setOutputEnable(object->hPin, hwAttrs->txPin, 1);
400  PIN_setOutputValue(object->hPin, hwAttrs->txPin, 1);
401  /* 2. Disconntect tx from IO, and set it as "GPIO" */
402  PINCC26XX_setMux(object->hPin, hwAttrs->txPin, IOC_PORT_GPIO);
403  /* 3. Disconnect cts */
404  PINCC26XX_setMux(object->hPin, hwAttrs->ctsPin, IOC_PORT_GPIO);
405  /*4. Wait for TX FIFO to become empty.
406  * CALLBACK: Idle until the TX FIFO is empty, i.e. no longer busy.
407  * BLOCKING: Periodically check if TX is busy emptying the FIFO.
408  * Must be handled at TX FIFO empty clock timeout:
409  * - the timeout/finish function must check the status
410  */
411  if(object->writeMode == UART_MODE_CALLBACK) {
412  /* Wait until the TX FIFO is empty. CALLBACK mode can be used from
413  * hwi/swi context, so we cannot use semaphore..
414  */
415  while(UARTBusy(hwAttrs->baseAddr));
416  } else { /* i.e. UART_MODE_BLOCKING */
417  /* Pend on semaphore again..(this time forever since we are flushing
418  * TX FIFO and nothing should be able to stop it..
419  */
420  Semaphore_pend(Semaphore_handle(&(object->writeSem)), BIOS_WAIT_FOREVER);
421  }
422  /* 5. Revert to active pins before returning */
423  PINCC26XX_setMux(object->hPin, hwAttrs->txPin, IOC_PORT_MCU_UART0_TX);
424  if(isFlowControlEnabled(hwAttrs)) {
425  PINCC26XX_setMux(object->hPin, hwAttrs->ctsPin, IOC_PORT_MCU_UART0_CTS);
426  }
427 }

429 /*
430  * ======== UARTCC26XX_hwiIntFxn ========
431  * Hwi function that processes UART interrupts.
432  *
433  * Six UART interrupts are enabled when configured for RX: Receive timeout,
434  * receive FIFO is 4/8 full and all four error interrupts.
435  *
436  * One interrupt is enabled when configured for TX: Transmit FIFO is 7/8
437  * empty.
438  *
439  * The RX and TX can operate independently of each other.
440  *
441  * When the read or write is finished they will
442  * post the semaphore or make the callback and log the transfer.
443  *
444  * @param arg The UART_Handle for this Hwi.
445  */
446 void UARTCC26XX_hwiIntFxn(UArg arg)
447 {
448  unsigned long intStatus;
449  unsigned long errStatus = UART_OK;
450  UARTCC26XX_Object *object;
451  UARTCC26XX_HWAttrsV2 const *hwAttrs;
452 
453  /* Get the pointer to the object and hwAttrs */
454  object = ((UART_Handle)arg)->object;
455  hwAttrs = ((UART_Handle)arg)->hwAttrs;
456 
457  /* Clear interrupts */
458  intStatus = UARTIntStatus(hwAttrs->baseAddr, true);
459  UARTIntClear(hwAttrs->baseAddr, intStatus);
460 
461  Log_print2(Diags_USER2, "UART:(%p) Interrupt with mask 0x%x",
462  hwAttrs->baseAddr, intStatus);
463 
464  /* Record readSize */
465  int32_t readSize = object->readSize;
466  int32_t readSize_ringBuf = object->ringBuffer.length - object->ringBuffer.count;
467 
468  /* Basic error handling */
469  if(intStatus & (UART_INT_OE | UART_INT_BE | UART_INT_PE | UART_INT_FE)) {
470 
471  /* If overrun, the error bit is set immediately */
472  if(intStatus & UART_INT_OE) {
473  /* check receive status register */
474  errStatus = UARTRxErrorGet(hwAttrs->baseAddr);
475  UARTRxErrorClear(hwAttrs->baseAddr);
476  /* read out min of readsize and fifo size */
477  int32_t bytesToRead = MIN(UARTCC26XX_FIFO_SIZE, readSize);
478  readData((UART_Handle)arg, bytesToRead);
479  /* read the rest into RingBuf */
480  bytesToRead = MIN((UARTCC26XX_FIFO_SIZE - bytesToRead), readSize_ringBuf);
481  readData2RingBuf((UART_Handle)arg, bytesToRead);
482  }
483  else {
484  /*
485  * else, for break, framing and par. error bits are at error index
486  * (we would like to keep the "surviving" bits)
487  */
488  /* Only if UART_read is active */
489  while(object->readSize > 0) {
490  /* read one data */
491  readData((UART_Handle)arg, 1);
492  /* check receive status register if byte is OK */
493  errStatus = UARTRxErrorGet(hwAttrs->baseAddr);
494  UARTRxErrorClear(hwAttrs->baseAddr);
495  if(errStatus != 0) {
496  /*
497  * last read was not including data, reset data vars
498  * (i.e. upd readBuf pointer, decr readCount)
499  */
500  object->readBuf = (unsigned char *)object->readBuf - 1;
501  object->readCount--;
502  /* the FIFO index with error is reached, stop reading more. */
503  break;
504  }
505  else {
506  /* current read was not the problem update readSize */
507  object->readSize--;
508  }
509  }
510  }
511 
512  /* Report current error status */
513  object->status = (UART_Status)errStatus;
514 
515  /* Break and clean up any ongoing transaction */
516  UARTCC26XX_readCancel((UART_Handle)arg);
517  }
518  else {
519  /* Receive Timeout */
520  if(intStatus & UART_INT_RT) {
521  /* if UART_read is active */
522  if (readSize) {
523  /* Read data from FIFO. */
524  object->readSize = readData((UART_Handle)arg, readSize);
525 
526  /* If return partial is set */
527  if(object->readRetPartial) {
528  /* Partial read accepted, read succeeded by def, set readSize
529  * to allow callback to trigger new _read()
530  */
531  object->readSize = 0;
532  /* Read succeeded */
533  Swi_post(Swi_handle(&(object->swi)));
534  }
535  /* else - return when all bytes have arrived */
536  else {
537  /* If all bytes are read */
538  if(!object->readSize) {
539  /* Read succeeded */
540  Swi_post(Swi_handle(&(object->swi)));
541  }
542  }
543  }
544  else {
545  /* not during UART_read, put in RingBuf */
546  readData2RingBuf((UART_Handle)arg, readSize_ringBuf);
547  }
548  }
549  else {
550  /* RX interrupt, since CTS is handled by HW */
551  if (intStatus & UART_INT_RX) {
552  if (readSize) {
553  /* if UART_read is active */
554  /* Read whatever is less of:
555  * - FIFO_THR bytes - 1 (leave one byte to trigger RT timeout)
556  * - object->readSize (never read more than requested from application)
557  * since RT timeout will only trigger if FIFO is not empty.
558  */
559  int32_t bytesToRead = MIN(object->readFifoThreshold-1, object->readSize);
560 
561  /* Do read, all bytes we request to read, are present in FIFO */
562  readData((UART_Handle)arg, bytesToRead);
563 
564  /* Decrement object->readSize with the actual number of bytes read
565  * There will always be at least bytesToRead number of bytes in FIFO
566  */
567  object->readSize -= bytesToRead;
568  /* If all bytes are read */
569  if (!object->readSize) {
570  /* Read succeeded. */
571  Swi_post(Swi_handle(&(object->swi)));
572  }
573  }
574  else {
575  /* not during UART_read, put all in RingBuf */
576  readData2RingBuf((UART_Handle)arg, readSize_ringBuf);
577  }
578  }
579  }
580  }
581 
582  /* Write if there are characters to be written. */
583  if ((intStatus & UART_INT_TX) && object->writeSize) {
584  /* Using writtenLast=writeSize before last write
585  *(since writeSize=0 when the last finishes)
586  */
587  uint32_t writtenLast = object->writeSize;
588  object->writeSize = writeData((UART_Handle)arg, object->writeSize);
589  if(!object->writeSize) {
590  /* No more to write, but data is not shiftet out properly yet.
591  * Start TX FIFO Empty clock, which will trigger the txFifoEmptyClk
592  * function.
593  * +UART_TH_FIFO_1_8 because it is 4 bytes left in TX FIFO when the TX FIFO
594  * threshold interrupt occurs.
595  */
596  startTxFifoEmptyClk((UART_Handle)arg, (writtenLast+UART_TH_FIFO_1_8));
597  }
598  }
599 }

603  * Swi function that processes UART interrupts.
604  * @param arg The UART_Handle for this Hwi.
605  */
606 void UARTCC26XX_swiIntFxn(UArg arg0, UArg arg1)

609

613  /* Release power constraint. */
